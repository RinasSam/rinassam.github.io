38 static int ParseParams(int argc, char *argv[])
39 {
40  int i, new_thpri_hi, new_thpri_lo;
41  const char *param;
42 
43  new_thpri_hi = PADMAN_THPRI_HI;
44  new_thpri_lo = PADMAN_THPRI_LO;
45 
46  if(argc > 1)
47  {
48  argv++;
49 
50  for(i = 1; i < argc; i++,argv++)
51  {
52  if(strncmp("thpri=", *argv, 6) == 0)
53  {
54  //Parse high priority
55  param = &(*argv)[6];
56  if(isdigit(*param))
57  new_thpri_hi = strtol(param, NULL, 10);
58 
59  //Skip value for high priority
60  while(isdigit(*param))
61  param++;
62 
63  //Valid high priority parameter
64  if(new_thpri_hi - 9 < 115)
65  {
66  if(*param == ',')
67  {
68  ++param;
69  //Parse low priority
70  if(isdigit(*param))
71  new_thpri_lo = strtol(param, NULL, 10);
72 
73  if(new_thpri_lo - 9 >= 115)
74  {
75  M_KPRINTF("invalid priority %d\n", new_thpri_lo);
76  return 0;
77  }
78  }
79 
80  if(new_thpri_lo < new_thpri_hi)
81  M_PRINTF("high prio thread must be higher than low prio thread\n");
82  }
83  else
84  {
85  M_KPRINTF("invalid priority %d\n", new_thpri_hi);
86  return 0;
87  }
88  }
89  }
90  }
91 
92  thpri_hi = new_thpri_hi;
93  thpri_lo = new_thpri_lo;
94 
95  return 1;
96 }
