74  ILINKRegisterBase->PHT_ReqResHdr0_R0=(((unsigned int)trContext->NodeID)<<16) | offset_high;

84 
85  DEBUG_PRINTF("Waiting for interrupt...\n");

89 
90  DEBUG_PRINTF("Processing results...\n");
