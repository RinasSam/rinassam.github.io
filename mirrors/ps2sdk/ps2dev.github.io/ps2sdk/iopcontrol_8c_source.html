46  if(arg != NULL)
47  {

53  arglen = 0;
54  }

59  reset_pkt.mode = mode;

63  dmat.size = sizeof(reset_pkt);

65  SifWriteBackDCache(&reset_pkt, sizeof(reset_pkt));
