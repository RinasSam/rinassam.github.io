146 static int fio_read(iop_file_t *f, void *buffer, int size)
147 {
148  int i;
149 
150  int start_sector;
151  int off_sector;
152  int num_sectors;
153 
154  int read = 0;
155  static u8 local_buffer[9 * 2048];
156 
157  DPRINTF("CDFS: fio_read called\n\n");
158  DPRINTF(" kernel_fd... %p\n", f);
159  DPRINTF(" buffer...... 0x%X\n", (int)buffer);
160  DPRINTF(" size........ %d\n\n", size);
161 
162  i = (int)f->privdata;
163 
164  if (i >= MAX_FILES_OPENED) {
165  printf("fio_read: ERROR: File does not appear to be open!\n");
166  return -1;
167  }
168 
169  // A few sanity checks
170  if (fd_table[i].filePos > fd_table[i].fileSize) {
171  // We cant start reading from past the beginning of the file
172  return 0; // File exists but we couldnt read anything from it
173  }
174 
175  if ((fd_table[i].filePos + size) > fd_table[i].fileSize)
176  size = fd_table[i].fileSize - fd_table[i].filePos;
177 
178  if (size <= 0)
179  return 0;
180 
181  if (size > MAX_BYTES_READ)
182  size = MAX_BYTES_READ;
183 
184  // Now work out where we want to start reading from
185  start_sector = fd_table[i].LBA + (fd_table[i].filePos >> 11);
186  off_sector = (fd_table[i].filePos & 0x7FF);
187 
188  num_sectors = (off_sector + size);
189  num_sectors = (num_sectors >> 11) + ((num_sectors & 2047) != 0);
190 
191  DPRINTF("fio_read: read sectors %d to %d\n", start_sector, start_sector + num_sectors);
192 
193  // Skip a Sector for equal (use the last sector in buffer)
194  if (start_sector == lastsector) {
195  read = 1;
196  if (last_bk > 0)
197  memcpy(local_buffer, local_buffer + 2048 * (last_bk), 2048);
198  last_bk = 0;
199  }
200 
201  lastsector = start_sector + num_sectors - 1;
202  // Read the data (we only ever get 16KB max request at once)
203 
204  if (read == 0 || (read == 1 && num_sectors > 1)) {
205  if (!cdfs_readSect(start_sector + read, num_sectors - read, local_buffer + ((read) << 11))) {
206  DPRINTF("Couldn't Read from file for some reason\n");
207  }
208 
209  last_bk = num_sectors - 1;
210  }
211 
212  memcpy(buffer, local_buffer + off_sector, size);
213  fd_table[i].filePos += size;
214 
215  return (size);
216 }
