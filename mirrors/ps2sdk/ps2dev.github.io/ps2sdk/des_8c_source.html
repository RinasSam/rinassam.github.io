401  40 41 42 43 44 45 46 47 61 53 45 37 29 21 13 5
402  48 49 50 51 52 53 54 55 59 51 43 35 27 19 11 3
403  56 57 58 59 60 61 62 63 57 49 41 33 25 17 9 1
404 
405  The output has been subject to swaps of the form
406  0 1 -> 3 1 but the odd and even bits have been put into
407  2 3 2 0
408  different words. The main trick is to remember that
409  t=((l>>size)^r)&(mask);
410  r^=t;
411  l^=(t<<size);
412  can be used to swap and move bits between words.
413 
414  So l = 0 1 2 3 r = 16 17 18 19
415  4 5 6 7 20 21 22 23
416  8 9 10 11 24 25 26 27
417  12 13 14 15 28 29 30 31
418  becomes (for size == 2 and mask == 0x3333)
419  t = 2^16 3^17 -- -- l = 0 1 16 17 r = 2 3 18 19
420  6^20 7^21 -- -- 4 5 20 21 6 7 22 23
421  10^24 11^25 -- -- 8 9 24 25 10 11 24 25
422  14^28 15^29 -- -- 12 13 28 29 14 15 28 29

425  could be done in 15 xor, 10 shifts and 5 ands.
426  When I finally started to think of the problem in 2D
427  I first got ~42 operations without xors. When I remembered
428  how to use xors :-) I got it to its final state.

486  PERM_OP (d,c,t,1,0x55555555);

525 unsigned char *DES(const unsigned char *key, const unsigned char *message, unsigned char *cipher)
526 {
527  unsigned int l, r, t, u;
528  int i;
529  unsigned int *s;
530  unsigned char *keys;
531 
532  keys = DES_createkeys(key);
533 
534  c2l(message, l); /* get endian free int from input block */
535  c2l(message, r); /* get endian free int from input block */
536 
537  /* do IP */
538  PERM_OP(r,l,t, 4,0x0f0f0f0f);
539  PERM_OP(l,r,t,16,0x0000ffff);
540  PERM_OP(r,l,t, 2,0x33333333);
541  PERM_OP(l,r,t, 8,0x00ff00ff);
542  PERM_OP(r,l,t, 1,0x55555555);
543  /* r and l are reversed - remember that :-) - fix
544  * it in the next step */
545 
546  /* Things have been modified so that the initial rotate is
547  * done outside the loop. This required the
548  * des_SPtrans values in sp.h to be rotated 1 bit to the right.
549  * One perl script later and things have a 5% speed up on a sparc2.
550  * Thanks to Richard Outerbridge <71755.204@CompuServe.COM>
551  * for pointing this out. */
552  t = (r << 1) | (r >> 31);
553  r = (l << 1) | (l >> 31);
554  l = t;
555 
556  s = (unsigned int *)keys;
557  /* I don't know if it is worth the effort of loop unrolling the
558  * inner loop */
559  for (i=0; i<32; i+=4) {
560  D_ENCRYPT(l,r,i+0); /* 1 */
561  D_ENCRYPT(r,l,i+2); /* 2 */
562  }
563  l = (l >> 1) | (l << 31);
564  r = (r >> 1) | (r << 31);
565 
566  /* swap l and r
567  * we will not do the swap so just remember they are
568  * reversed for the rest of the subroutine
569  * luckily FP fixes this problem :-) */
570 
571  PERM_OP(r,l,t, 1,0x55555555);
572  PERM_OP(l,r,t, 8,0x00ff00ff);
573  PERM_OP(r,l,t, 2,0x33333333);
574  PERM_OP(l,r,t,16,0x0000ffff);
575  PERM_OP(r,l,t, 4,0x0f0f0f0f);
576 
577  l2c(l, cipher); /* get endian free int from input block */
578  l2c(r, cipher); /* get endian free int from input block */
579 
580  return (unsigned char *)cipher;
581 }
