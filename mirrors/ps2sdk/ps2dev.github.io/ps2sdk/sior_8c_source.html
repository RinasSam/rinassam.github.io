114 int _start(int argc, char **argv)
115 {
116  int retries;
117 
118  memset(&cd0, 0, sizeof(cd0));
119 
120  for (retries = 0; retries < 15; retries++) {
121  printf("Binding RPC.\n");
122  if (SifBindRpc(&cd0, SIOR_IRX, 0) < 0) {
123  printf("Failed!\n");
124  return MODULE_NO_RESIDENT_END;
125  }
126  if (cd0.server != 0) {
127  printf("Registering library.\n");
128  if (RegisterLibraryEntries(&_exp_sior)) {
129  printf("Couldn't register.\n");
130  return MODULE_NO_RESIDENT_END;
131  }
132  return MODULE_RESIDENT_END;
133  }
134  printf("Huh, not available, retrying... ?\n");
135  DelayThread(1000);
136  }
137 
138  printf("Giving up.\n");
139 
140  return MODULE_NO_RESIDENT_END;
