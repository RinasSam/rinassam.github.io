404 int dev9DmaTransfer(int device, void *buf, int bcr, int dir)
405 {
406  USE_SPD_REGS;
407  volatile iop_dmac_chan_t *dev9_chan = (iop_dmac_chan_t *)DEV9_DMAC_BASE;
408  int res, dmactrl, OldState;
409 
410  if (device >= 2) {
411  if (dev9_predma_cbs[device] == NULL)
412  return -1;
413  if (dev9_postdma_cbs[device] == NULL)
414  return -1;
415  }
416 
417  if ((res = WaitSema(dma_lock_sem)) < 0)
418  return res;
419 
420  switch (device) {
421  case 0:
422  dmactrl = 0;
423  break;
424  case 1:
425  dmactrl = 1;
426  break;
427 #ifdef DEV9_PSX_SUPPORT
428  // Used for the PSX
429  case 2:
430  dmactrl = 0x10;
431  break;
432  case 3:
433  dmactrl = 0x20;
434  break;
435 #endif
436  default:
437  dmactrl = 0;
438  }
439 
440  SPD_REG16(SPD_R_DMA_CTRL) = (SPD_REG16(SPD_R_REV_1) < 17) ? (dmactrl & 0x03) | 0x04 : dmactrl | 0x06;
441 
442  if (dev9_predma_cbs[device])
443  dev9_predma_cbs[device](bcr, dir);
444 
445  dev9_chan->madr = (u32)buf;
446 
447  /* Older versions of DEV9 do not suspend interrupts. Not sure why this must be done though. */
448  CpuSuspendIntr(&OldState);
449  dev9_chan->bcr = bcr;
450  dev9_chan->chcr = DMAC_CHCR_30 | DMAC_CHCR_TR | DMAC_CHCR_CO | (dir & DMAC_CHCR_DR);
451  CpuResumeIntr(OldState);
452 
453  /* Wait for DMA to complete. Do not use a semaphore as thread switching hurts throughput greatly. */
454  while (dev9_chan->chcr & DMAC_CHCR_TR) {}
455 
456  if (dev9_postdma_cbs[device])
457  dev9_postdma_cbs[device](bcr, dir);
458 
459  SignalSema(dma_lock_sem);
460  return 0;
461 }

463 static int read_eeprom_data(void)
464 {
465  USE_SPD_REGS;
466  int i, j, res = -2;
467  u8 val;
468 
469  if (eeprom_data[0] < 0)
470  goto out;
471 
472  SPD_REG8(SPD_R_PIO_DIR) = 0xe0 | (dev9_has_dvr_capability ? 7 : 1);
473  DelayThread(1);
474  SPD_REG8(SPD_R_PIO_DATA) = 0x80;
475  DelayThread(1);
476 
477  for (i = 0; i < 2; i++) {
478  SPD_REG8(SPD_R_PIO_DATA) = 0xa0;
479  DelayThread(1);
480  SPD_REG8(SPD_R_PIO_DATA) = 0xe0;
481  DelayThread(1);
482  }
483  for (i = 0; i < 7; i++) {
484  SPD_REG8(SPD_R_PIO_DATA) = 0x80;
485  DelayThread(1);
486  SPD_REG8(SPD_R_PIO_DATA) = 0xc0;
487  DelayThread(1);
488  }
489  SPD_REG8(SPD_R_PIO_DATA) = 0xc0;
490  DelayThread(1);
491 
492  val = SPD_REG8(SPD_R_PIO_DATA);
493  DelayThread(1);
494  if (val & 0x10) { /* Error. */
495  SPD_REG8(SPD_R_PIO_DATA) = 0;
496  DelayThread(1);
497  res = -1;
498  eeprom_data[0] = 0;
499  goto out;
500  }
501 
502  SPD_REG8(SPD_R_PIO_DATA) = 0x80;
503  DelayThread(1);
504 
505  /* Read the MAC address and checksum from the EEPROM. */
506  for (i = 0; i < 4; i++) {
507  eeprom_data[i + 1] = 0;
508 
509  for (j = 15; j >= 0; j--) {
510  SPD_REG8(SPD_R_PIO_DATA) = 0xc0;
511  DelayThread(1);
512  val = SPD_REG8(SPD_R_PIO_DATA);
513  if (val & 0x10)
514  eeprom_data[i + 1] |= (1 << j);
515  SPD_REG8(SPD_R_PIO_DATA) = 0x80;
516  DelayThread(1);
517  }
518  }
519 
520  SPD_REG8(SPD_R_PIO_DATA) = 0;
521  DelayThread(1);
522  eeprom_data[0] = 1; /* The EEPROM data is valid. */
523  res = 0;
524 
525 out:
526  SPD_REG8(SPD_R_PIO_DIR) = dev9_has_dvr_capability ? 7 : 1;
527  return res;
528 }
