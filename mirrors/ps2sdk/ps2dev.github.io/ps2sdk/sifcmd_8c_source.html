57 static unsigned int _SifSendCmd(int cid, int mode, void *pkt, int pktsize, void *src,
58  void *dest, int size)
59 {
60  SifDmaTransfer_t dmat[2];
61  SifCmdHeader_t *header;
62  int count = 0;
63 
64  pktsize &= 0xff;
65 
66  if (pktsize > CMD_PACKET_DATA_MAX)
67  return 0;
68 
69  header = (SifCmdHeader_t *)pkt;
70  header->cid = cid;
71  header->psize = pktsize;
72  header->dest = NULL;
73 
74  if (size > 0) {
75  header->dsize = size;
76  header->dest = dest;
77 
78  if (mode & SIF_CMD_M_WBDC) /* if mode is & 4, flush reference cache */
79  SifWriteBackDCache(src, size);
80 
81  dmat[count].src = src;
82  dmat[count].dest = dest;
83  dmat[count].size = size;
84  dmat[count].attr = 0;
85  count++;
86  }
87 
88  dmat[count].src = pkt;
89  dmat[count].dest = _sif_cmd_data.iopbuf;
90  dmat[count].size = pktsize;
91  dmat[count].attr = SIF_DMA_ERT | SIF_DMA_INT_O;
92  count++;
93 
94  SifWriteBackDCache(pkt, pktsize);
95 
96  if (mode & SIF_CMD_M_INTR) /* INTERRUPT DMA TRANSFER */
97  return iSifSetDma(dmat, count);
98  else
99  return SifSetDma(dmat, count);
100 }

120 int _SifCmdIntHandler(int channel)
121 {
122  u128 packet[8];
123  u128 *pktbuf;
124  struct cmd_data *cmd_data = &_sif_cmd_data;
125  SifCmdHeader_t *header;
126  SifCmdHandlerData_t *cmd_handlers;
127  int size, pktquads, id, i = 0;
128 
129  EI();
130 
131  header = (SifCmdHeader_t *)cmd_data->pktbuf;
132 
133  if (!(size = header->psize))
134  goto out;
135 
136  pktquads = (size + 30) >> 4;
137  header->psize = 0;
138  if (pktquads) {
139  pktbuf = (u128 *)cmd_data->pktbuf;
140  while (pktquads--) {
141  packet[i] = pktbuf[i];
142  i++;
143  }
144  }
145 
146  iSifSetDChain();
147 
148  header = (SifCmdHeader_t *)packet;
149  /* Get the command handler id and determine which handler list to
150  dispatch from. */
151  id = header->cid & ~SIF_CMD_ID_SYSTEM;
152 
153  if (header->cid & SIF_CMD_ID_SYSTEM) {
154  if (id < cmd_data->nr_sys_handlers)
155  cmd_handlers = cmd_data->sys_cmd_handlers;
156  else
157  goto out;
158  }
159  else {
160  if (id < cmd_data->nr_usr_handlers)
161  cmd_handlers = cmd_data->usr_cmd_handlers;
162  else
163  goto out;
164  }
165 
166  if (cmd_handlers[id].handler)
167  cmd_handlers[id].handler(packet, cmd_handlers[id].harg);
168 
169 out:
170  ExitHandler();
171  return 0;
172 }

215 void SifInitCmd(void)
216 {
217  static struct ca_pkt packet __attribute((aligned(64)));
218  int i;
219  static int _rb_count = 0;
220  if(_rb_count != _iop_reboot_count)
221  {
222  _rb_count = _iop_reboot_count;
223  if (sif0_id >= 0)
224  {
225  DisableDmac(DMAC_SIF0);
226  RemoveDmacHandler(DMAC_SIF0, sif0_id);
227  }
228  init = 0;
229  }
230 
231  if (init)
232  return;
233 
234  DI();
235 
236  _sif_cmd_data.pktbuf = UNCACHED_SEG(pktbuf);
237  _sif_cmd_data.unused = UNCACHED_SEG(sif_unused);
238  _sif_cmd_data.sys_cmd_handlers = sys_cmd_handlers;
239  _sif_cmd_data.nr_sys_handlers = SYS_CMD_HANDLER_MAX;
240  _sif_cmd_data.usr_cmd_handlers = NULL;
241  _sif_cmd_data.nr_usr_handlers = 0;
242  _sif_cmd_data.sregs = sregs;
243 
244  for (i = 0; i < SYS_CMD_HANDLER_MAX; i++) {
245  _sif_cmd_data.sys_cmd_handlers[i].handler = NULL;
246  _sif_cmd_data.sys_cmd_handlers[i].harg = NULL;
247  }
248 
249  for (i = 0; i < 32; i++)
250  _sif_cmd_data.sregs[i] = 0;
251 
252  _sif_cmd_data.sys_cmd_handlers[0].handler = change_addr;
253  _sif_cmd_data.sys_cmd_handlers[0].harg = &_sif_cmd_data;
254  _sif_cmd_data.sys_cmd_handlers[1].handler = set_sreg;
255  _sif_cmd_data.sys_cmd_handlers[1].harg = &_sif_cmd_data;
256 
257  EI();
258  FlushCache(0);
259 
260  if (_lw(DMAC_COMM_STAT) & STAT_SIF0)
261  _sw(STAT_SIF0, DMAC_COMM_STAT);
262 
263  //If SIF0 (IOP -> EE) is not enabled, enable it.
264  if (!(_lw(DMAC_SIF0_CHCR) & CHCR_STR))
265  SifSetDChain();
266 
267  sif0_id = AddDmacHandler(DMAC_SIF0, &_SifCmdIntHandler, 0);
268  EnableDmac(DMAC_SIF0);
269 
270  init = 1;
271 
272  _sif_cmd_data.iopbuf = (void *)SifGetReg(SIF_SYSREG_SUBADDR);
273  if (_sif_cmd_data.iopbuf) {
274  /* IOP SIF CMD is already initialized, so give it our new
275  receive address. */
276  packet.buf = _sif_cmd_data.pktbuf;
277  SifSendCmd(SIF_CMD_CHANGE_SADDR, &packet, sizeof packet, NULL, NULL, 0);
278  } else {
279  /* Sync with the IOP side's SIFCMD implementation. */
280  while (!(SifGetReg(SIF_REG_SMFLAG) & SIF_STAT_CMDINIT)) ;
281 
282  _sif_cmd_data.iopbuf = (void *)SifGetReg(SIF_REG_SUBADDR);
283  SifSetReg(SIF_SYSREG_SUBADDR, (u32)_sif_cmd_data.iopbuf);
284  /* See the note above about struct cmd_data, and the use of
285  this register. */
286  SifSetReg(SIF_SYSREG_MAINADDR, (u32)&_sif_cmd_data);
287  packet.header.opt = 0; //SIFCMD initialization on the IOP (opt = 0)
288  packet.buf = _sif_cmd_data.pktbuf;
289  SifSendCmd(SIF_CMD_INIT_CMD, &packet, sizeof packet, NULL, NULL, 0);
290  }
291 }
