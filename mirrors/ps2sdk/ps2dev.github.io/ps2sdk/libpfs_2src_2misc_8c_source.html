65 int pfsGetTime(pfs_datetime_t *tm)
66 {
67 #ifdef _IOP
68  int ret, i;
69  sceCdCLOCK cdtime;
70  static pfs_datetime_t timeBuf={
71  0, 7, 6, 5, 4, 3, 2000 // used if can not get time...
72  };
73 
74  for(i = 0; i < 20; i++)
75  {
76  ret = sceCdReadClock(&cdtime);
77 
78  if(ret!=0 && cdtime.stat==0)
79  {
80  timeBuf.sec=btoi(cdtime.second);
81  timeBuf.min=btoi(cdtime.minute);
82  timeBuf.hour=btoi(cdtime.hour);
83  timeBuf.day=btoi(cdtime.day);
84  timeBuf.month=btoi(cdtime.month & 0x7F); //The old CDVDMAN sceCdReadClock() function does not automatically file off the highest bit.
85  timeBuf.year=btoi(cdtime.year) + 2000;
86  break;
87  } else {
88  if(!(cdtime.stat & 0x80))
89  break;
90  }
91 
92  DelayThread(100000);
93  }
94  memcpy(tm, &timeBuf, sizeof(pfs_datetime_t));
95 #else
96  time_t rawtime;
97  struct tm timeinfo;
98  time(&rawtime);
99  // Convert to JST
100  rawtime += (-9 * 60 * 60);
101  gmtime_r(&rawtime, &timeinfo);
102 
103  tm->sec = timeinfo.tm_sec;
104  tm->min = timeinfo.tm_min;
105  tm->hour = timeinfo.tm_hour;
106  tm->day = timeinfo.tm_mday;
107  tm->month = timeinfo.tm_mon + 1;
108  tm->year = timeinfo.tm_year + 1900;
109 #endif
110 
111  return 0;
112 }

114 int pfsFsckStat(pfs_mount_t *pfsMount, pfs_super_block_t *superblock,
115  u32 stat, int mode)
116 { // mode 0=set flag, 1=remove flag, else check stat
117 
118  if(pfsMount->blockDev->transfer(pfsMount->fd, superblock, 0, PFS_SUPER_SECTOR, 1,
119  PFS_IO_MODE_READ)==0)
120  {
121  switch(mode)
122  {
123  case PFS_MODE_SET_FLAG:
124  superblock->pfsFsckStat|=stat;
125  break;
126  case PFS_MODE_REMOVE_FLAG:
127  superblock->pfsFsckStat&=~stat;
128  break;
129  default/*PFS_MODE_CHECK_FLAG*/:
130  return 0 < (superblock->pfsFsckStat & stat);
131  }
132  pfsMount->blockDev->transfer(pfsMount->fd, superblock, 0, PFS_SUPER_SECTOR, 1,
133  PFS_IO_MODE_WRITE);
134  pfsMount->blockDev->flushCache(pfsMount->fd);
135  }
136  return 0;
137 }

212 pfs_block_device_t *pfsGetBlockDeviceTable(const char *name)
213 {
214  char *end;
215  char devname[32];
216  char *tmp;
217  u32 len;
218  int i;
219 
220  while(name[0] == ' ')
221  name++;
222 
223  end = strchr(name, ':');
224  if(!end) {
225  PFS_PRINTF(PFS_DRV_NAME": Error: Unknown block device '%s'\n", name);
226  return NULL;
227  }
228 
229  len = (u32)end - (u32)name;
230  strncpy(devname, name, len);
231  devname[len] = '\0';
232 
233  // Loop until digit is found, then terminate string at that digit.
234  // Should then have just the device name left, minus any front spaces or trailing digits.
235  tmp = devname;
236  while(!(isdigit(tmp[0])))
237  tmp++;
238  tmp[0] = '\0';
239 
240  for(i = 0; i < NUM_SUPPORTED_DEVICES; i++)
241  if(!strcmp(pfsBlockDeviceCallTable[i].devName, devname))
242  return &pfsBlockDeviceCallTable[i];
243 
244  return NULL;
245 }
