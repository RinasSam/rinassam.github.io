86 static void Shutdown(void* data)
87 {
88 #ifdef DEBUG
89  printf("Shutdown\n");
90 #endif
91  int i;
92  /* Do callbacks in reverse order */
93  for(i = MAX_CALLBACKS-1; i >= 0; i--)
94  {
95  if(CallbackTable[i].cb)
96  {
97  CallbackTable[i].cb(CallbackTable[i].data);
98  }
99  }
100 
101  // Turn off PS2
102  *((unsigned char *)0xBF402017) = 0;
103  *((unsigned char *)0xBF402016) = 0xF;
104 }

196 void* poweroff_rpc_server(int fno, void *data, int size)
197 {
198  switch(fno) {
199  case PWROFF_SHUTDOWN:
200  Shutdown(0);
201  break;
202 
203  case PWROFF_ENABLE_AUTO_SHUTOFF:
204  InitPowerOffThread();
205 
206  int* sbuff = data;
207  if (sbuff[0])
208  SetPowerButtonHandler(Shutdown, 0);
209  else
210  SetPowerButtonHandler(SendCmd, 0);
211  sbuff[0] = 1;
212  return sbuff;
213  }
214  return NULL;
215 }

226 int _start(int argc, char* argv[])
227 {
228  register struct handlerTableEntry *handlers=(struct handlerTableEntry*)0x480;//iopmem
229  iop_thread_t mythread;
230  int i;
231 
232  if(RegisterLibraryEntries(&_exp_poweroff) != 0)
233  {
234  printf("Poweroff already registered\n");
235  return 1;
236  }
237 
238  SetPowerButtonHandler(Shutdown, 0);
239 
240  if (handlers[IOP_IRQ_CDVD].handler==0) {
241  printf("No CDROM handler. Run CDVDMAN first\n");
242  return 1;
243  }
244 
245  if (((int)handlers[IOP_IRQ_CDVD].handler & 3) != TYPE_C){
246  printf("Cannot chain to non-C handler\n");
247  return 1;
248  }
249 
250  oldCdHandler=(intrhandler)((int)handlers[IOP_IRQ_CDVD].handler & ~3);
251  handlers[IOP_IRQ_CDVD].handler=(intrhandler)((int)myCdHandler | TYPE_C);
252 
253  memset(CallbackTable, 0, sizeof(struct CallbackEntry) * MAX_CALLBACKS);
254 
255  mythread.attr = TH_C;
256  mythread.option = PWROFF_IRX;
257  mythread.thread = poweroff_rpc_Thread;
258  mythread.stacksize = 0x1000;
259  mythread.priority = 0x27;
260 
261  int pid = CreateThread(&mythread);
262 
263  if (pid > 0) {
264  if ((i=StartThread(pid, NULL)) < 0) {
265  printf("StartThread failed (%d)\n", i);
266  return MODULE_NO_RESIDENT_END;
267  }
268  }
269  else {
270  printf("CreateThread failed (%d)\n", pid);
271  return MODULE_NO_RESIDENT_END;
272  }
273 
274  printf("Poweroff installed\n");
275  return MODULE_RESIDENT_END;
276 }
