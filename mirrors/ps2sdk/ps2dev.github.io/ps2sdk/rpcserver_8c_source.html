176 static void* RpcServer(int fno, void *buffer, int length)
177 {
178  u32 *data = (u32*)buffer;
179  u32 command = data[0];
180 
181  if ((0x00000100 & command) != 0)
182  {
183  if ((0x80000000 & command) != 0)
184  {
185  command &= 0xff;
186  command += 1;
187  }
188  else if (command == 0x00000100)
189  {
190  // Unofficial: libpad EE client from v1.3.4 has this RPC function implemented, but is not implemented within its PADMAN module.
191  return buffer;
192  }
193  }
194 
195  switch(command)
196  {
197  case PAD_RPCCMD_INIT: return RpcPadInit(data);
198  case PAD_RPCCMD_END: return RpcPadEnd(data);
199  case PAD_RPCCMD_GET_MODVER: return RpcGetModVersion(data);
200  case PAD_RPCCMD_OPEN: return RpcPadOpen(data);
201  case PAD_RPCCMD_CLOSE: return RpcPadClose(data);
202  case PAD_RPCCMD_INFO_ACT: return RpcPadInfoAct(data);
203  case PAD_RPCCMD_INFO_COMB: return RpcPadInfoComb(data);
204  case PAD_RPCCMD_INFO_MODE: return RpcPadInfoMode(data);
205  case PAD_RPCCMD_SET_MMODE: return RpcPadSetMainMode(data);
206  case PAD_RPCCMD_SET_ACTDIR: return RpcPadSetActDirect(data);
207  case PAD_RPCCMD_SET_ACTALIGN: return RpcPadSetActAlign(data);
208  case PAD_RPCCMD_GET_BTNMASK: return RpcPadGetButtonMask(data);
209  case PAD_RPCCMD_SET_BTNINFO: return RpcPadSetButtonInfo(data);
210  case PAD_RPCCMD_SET_VREF: return RpcPadSetVrefParam(data);
211  case PAD_RPCCMD_GET_PORTMAX: return RpcPadGetPortMax(data);
212  case PAD_RPCCMD_GET_SLOTMAX: return RpcPadGetSlotMax(data);
213 
214  default:
215  M_PRINTF("invalid function code (%03x)\n", (int)data[0]);
216  break;
217  }
218 
219  return buffer;
220 }
