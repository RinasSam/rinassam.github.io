145 unsigned char *MD4(unsigned char *message, int len, unsigned char *cipher)
146 {
147  union {
148  unsigned char ucbuff[128];
149  unsigned int uibuff[128/sizeof(unsigned int)];
150  } buffer;
151  unsigned int b = len * 8;
152 
153  engineReset();
154 
155  while (len > 64) {
156  memcpy(buffer.ucbuff, message, 64);
157  transform(buffer.ucbuff);
158  message += 64;
159  len -= 64;
160  }
161 
162  memset(buffer.ucbuff, 0, 128);
163  memcpy(buffer.ucbuff, message, len);
164  buffer.ucbuff[len] = 0x80;
165 
166  if (len < 56) {
167  buffer.uibuff[56/sizeof(unsigned int)] = b;
168  transform(buffer.ucbuff);
169  } else {
170  buffer.uibuff[120/sizeof(unsigned int)] = b;
171  transform(buffer.ucbuff);
172  transform(&buffer.ucbuff[64]);
173  }
174 
175  *((unsigned int *)&cipher[0]) = context[0];
176  *((unsigned int *)&cipher[4]) = context[1];
177  *((unsigned int *)&cipher[8]) = context[2];
178  *((unsigned int *)&cipher[12]) = context[3];
179 
180  return (unsigned char *)cipher;
181 }
