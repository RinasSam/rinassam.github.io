60 int poweroffInit(void)
61 {
62  ee_thread_t thread;
63  int res;
64  static int _init_count = -1;
65 
66  if (_init_count == _iop_reboot_count)
67  return 0;
68  _init_count = _iop_reboot_count;
69 
70  while (((res = SifBindRpc(&cd0, PWROFF_IRX, 0)) < 0) || (cd0.server == NULL))
71  nopdelay();
72 
73  // Terminate and delete any previously created threads
74  if (powerOffThreadId >= 0)
75  {
76  TerminateThread(powerOffThreadId);
77  DeleteThread(powerOffThreadId);
78  SifRemoveRpc(&cb_srv, &cb_queue);
79  SifRemoveRpcQueue(&cb_queue);
80  powerOffThreadId = -1;
81  }
82 
83  thread.initial_priority = POWEROFF_THREAD_PRIORITY;
84  thread.stack_size = sizeof(poffThreadStack);
85  thread.gp_reg = &_gp;
86  thread.func = PowerOffThread;
87  thread.stack = (void *)poffThreadStack;
88  thread.option = 0;
89  thread.attr = 0;
90  powerOffThreadId = CreateThread(&thread);
91  StartThread(powerOffThreadId, NULL);
92 
93  int autoShutdown[4] = {0};
94  SifCallRpc(&cd0, PWROFF_ENABLE_AUTO_SHUTOFF, 0, NULL, 0, autoShutdown, sizeof(autoShutdown), NULL, NULL);
95 
96  return res;
97 }
