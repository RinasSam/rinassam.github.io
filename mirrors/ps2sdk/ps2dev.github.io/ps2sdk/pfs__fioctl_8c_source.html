70  return -ENODEV;

73  {

77 #endif

122 int pfsFioIoctl2(iop_file_t *f, int cmd, void *arg, size_t arglen, void *buf, size_t buflen)
123 {
124  int rv;
125  pfs_file_slot_t *fileSlot = (pfs_file_slot_t *)f->privdata;
126  pfs_mount_t *pfsMount;
127 
128  if(f->mode & O_DIROPEN)
129  if(cmd==PIOCATTRREAD)
130  return -EISDIR;
131 
132  if(!(f->mode & O_WRONLY))
133  {
134  switch(cmd)
135  {
136  case PIOCATTRLOOKUP:
137  case PIOCATTRREAD:
138  case PIOCINVINODE:
139  break;
140  default:
141  return -EACCES;
142  }
143  }
144  if((rv=pfsFioCheckFileSlot(fileSlot))<0)
145  return rv;
146  pfsMount=fileSlot->clink->pfsMount;
147 #ifdef PFS_SUPPORT_BHDD
148  if (strcmp(pfsMount->blockDev->devName, "bhdd") == 0)
149  return -EBADF;
150 #endif
151  switch(cmd)
152  {
153  case PIOCALLOC:
154  rv=pfsAllocZones(fileSlot->clink, *(int *)(arg), 1);
155  break;
156 
157  case PIOCFREE:
158  pfsFreeZones(fileSlot->clink);
159  break;
160 
161  case PIOCATTRADD:
162  case PIOCATTRDEL:
163  case PIOCATTRLOOKUP:
164  case PIOCATTRREAD:
165  rv=ioctl2Attr(fileSlot->clink, cmd, arg, buf, &fileSlot->aentryOffset);
166  break;
167 
168 #ifdef PFS_IOCTL2_INC_CHECKSUM
169  case PIOCINVINODE:
170  rv=ioctl2InvalidateInode(fileSlot->clink);
171  break;
172 #endif
173 
174  default:
175  rv=-EINVAL;
176  break;
177  }
178 
179  if(pfsMount->flags & PFS_FIO_ATTR_WRITEABLE)
180  pfsCacheFlushAllDirty(pfsMount);
181  rv=pfsFioCheckForLastError(pfsMount, rv);
182  SignalSema(pfsFioSema);
183 
184  return rv;
185 }

250 static pfs_aentry_t *getAentry(pfs_cache_t *clink, char *key, char *value, int mode)
251 {
252  int kLen, fullsize;
253  pfs_aentry_t *aentry=clink->u.aentry;
254  pfs_aentry_t *aentryLast=NULL;
255  pfs_aentry_t *end;
256 
257  kLen=strlen(key);
258  fullsize=(kLen+strlen(value)+7) & ~3;
259  for(end=(pfs_aentry_t *)((u8*)aentry+1024);aentry < end; aentry=(pfs_aentry_t *)((u8*)aentry+aentry->aLen))
260  { //Other than critical errors, do nothing about the filesystem errors.
261  if(aentry->aLen & 3)
262  PFS_PRINTF(PFS_DRV_NAME" Error: aentry allocated length/4 != 0\n");
263  if(aentry->aLen < ((aentry->kLen+aentry->vLen+7) & ~3))
264  {
265  PFS_PRINTF(PFS_DRV_NAME" Panic: aentry is too small\n");
266  return NULL;
267  }
268  if(end < (pfs_aentry_t*)((u8*)aentry+aentry->aLen))
269  PFS_PRINTF(PFS_DRV_NAME" Error: aentry too big\n");
270 
271  switch(mode)
272  {
273  case PFS_AENTRY_MODE_LOOKUP:
274  if(kLen==aentry->kLen)
275  if(memcmp(key, aentry->str, kLen)==0)
276  return aentry;
277  break;
278 
279  case PFS_AENTRY_MODE_ADD:
280  if(aentry->kLen==0)
281  {
282  if(aentry->aLen>=fullsize)
283  return aentry;
284  }
285  if(aentry->aLen - ((aentry->kLen+aentry->vLen+7) & ~3) < fullsize)
286  continue;
287  return aentry;
288 
289  case PFS_AENTRY_MODE_DELETE:
290  default:
291  if(kLen==aentry->kLen)
292  {
293  if(memcmp(key, aentry->str, kLen)==0)
294  {
295  if(aentryLast!=NULL)
296  {
297  aentryLast->aLen+=aentry->aLen;
298  return aentry;
299  }
300  // delete it :P
301  aentry->kLen=0;
302  aentry->vLen=0;
303  return aentry;
304  }
305  }
306  aentryLast=aentry;
307  break;
308 
309  }
310  }
311  return NULL;
312 }

314 static int ioctl2AttrAdd(pfs_cache_t *clink, pfs_ioctl2attr_t *attr)
315 {
316  u32 kLen, vLen;
317  pfs_aentry_t *aentry;
318  u32 tmp;
319 
320  // input check
321  kLen=strlen(attr->key);
322  vLen=strlen(attr->value);
323  if(kLen>=PFS_AENTRY_KEY_MAX || vLen>=PFS_AENTRY_VALUE_MAX) // max size bounds check
324  return -EINVAL;
325 
326  if(kLen==0 || vLen==0) // no input check
327  return -EINVAL;
328 
329  if(getAentry(clink, attr->key, NULL, PFS_AENTRY_MODE_LOOKUP))
330  return -EEXIST;
331  if((aentry=getAentry(clink, attr->key, attr->value, PFS_AENTRY_MODE_ADD)) == NULL)
332  return -ENOSPC;
333 
334  if(aentry->kLen==0)
335  tmp=aentry->aLen;
336  else
337  tmp=aentry->aLen-((aentry->kLen+(aentry->vLen + 7)) & 0x3FC); //The only case that uses 0x3FC within the whole PFS driver.
338 
339  aentry->aLen-=tmp;
340  aentry = (pfs_aentry_t*)((u8 *)aentry + aentry->aLen);
341  aentry->kLen=kLen;
342  aentry->vLen=vLen;
343  aentry->aLen=tmp;
344  memcpy(&aentry->str[0], attr->key, aentry->kLen);
345  memcpy(&aentry->str[aentry->kLen], attr->value, aentry->vLen);
346  clink->flags|=PFS_CACHE_FLAG_DIRTY;
347 
348  return 0;
349 }

374 static int ioctl2AttrRead(pfs_cache_t *clink, pfs_ioctl2attr_t *attr, u32 *offset)
375 {
376  pfs_aentry_t *aentry;
377 
378  if(*offset >= 1024)
379  return 0;
380  do {
381  aentry=(pfs_aentry_t *)((u8*)clink->u.inode+*offset);
382  memcpy(attr->key, &aentry->str[0], aentry->kLen);
383  attr->key[aentry->kLen]=0;
384  memcpy(attr->value, &aentry->str[aentry->kLen], aentry->vLen);
385  attr->value[aentry->vLen]=0;
386  *offset+=aentry->aLen; // next
387  if(aentry->kLen!=0)
388  break;
389  } while(*offset < 1024);
390 
391  return aentry->kLen;
392 }
