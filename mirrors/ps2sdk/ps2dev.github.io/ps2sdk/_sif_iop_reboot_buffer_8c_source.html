252 /* Generate an IOPRP image that contains IOPBTCONF, if the original contains IOPBTCONF.
253  This is required because UDNL will only seach succeeding IOPRP images for modules specified within IOPBTCONF. */
254 static int generateIOPBTCONF_img(void *output, const void *ioprp)
255 {
256  int size, offset, fsize_rounded;
257  romdir_t *romdir;
258  const u8 *ptr_in;
259  u8 *ptr_out;
260 
261  romdir = (romdir_t*)ioprp;
262  if(strcmp(romdir[0].name, "RESET") || strcmp(romdir[1].name, "ROMDIR"))
263  return -1;
264 
265  //Now locate IOPBTCONF
266  size = 0;
267  offset = 0;
268  for(; romdir->name[0] != '\0'; romdir++,offset += fsize_rounded)
269  {
270  fsize_rounded = (romdir->size + 15) & ~15;
271  if(strcmp(romdir->name, "IOPBTCONF") == 0)
272  {
273  romdir->name[0] = 'X'; //Change 'IOPBTCONF' to 'XOPBTCONF', so that UDNL will not find the original.
274  //Copy IOPBTCONF into the new image.
275  size = romdir->size + sizeof(iopbtconf_img_base);
276  ptr_out = (u8*)output;
277  ptr_in = (const u8*)ioprp;
278  memcpy(ptr_out, &iopbtconf_img_base, sizeof(iopbtconf_img_base));
279  memcpy(ptr_out + sizeof(iopbtconf_img_base), &ptr_in[offset], romdir->size);
280  ((romdir_t*)ptr_out)[3].size = romdir->size; //Update the size of IOPBTCONF within the generated IOPRP image.
281  break;
282  }
283  }
284 
285  return size;
286 }
