80 int fillbuf(int id, int chan)
81 {
82  u8 *buf = stream_buf;
83  int size;
84 
85  id &= 1;
86 
87  size = read(stream_fd, buf, stream_buflen);
88  if (size<0) /* Error */
89  {
90  dprintf(OUT_ERROR, "Channel%d: error: %d\n", chan, size);
91  return(-1);
92  }
93  if (size==0)
94  return(0); /* EOF */
95 
96  /* If we're stereo and we've read less than a chunk, we're screwed */
97  if ((stream_chans>1) && (size<stream_buflen))
98  {
99  dprintf(OUT_ERROR, "Channel%d: failed to read entire chunk (read %d bytes)\n", chan, size);
100  return(-1);
101  }
102 
103  setloopflags(id, buf, size);
104 
105  sceSdVoiceTrans(0, SD_TRANS_WRITE | SD_TRANS_MODE_DMA, buf, (u32*)(stream_bufspu[chan]+(id*stream_buflen)), size);
106  sceSdVoiceTransStatus(0, 1);
107 
108  return(size);
109 }

112 void stream_thread(void *a)
113 {
114  while(1)
115  {
116  /* Wait for an interrupt */
117  WaitSema(stream_sema);
118 
119  /* Update stuff */
120  stream_bufid = stream_bufsafe;
121  stream_cur+=(stream_buflen/16);
122 
123  dprintf(OUT_DEBUG, "SPU2 now playing buffer %d (block %d)\n", (int)stream_bufid, stream_cur);
124 
125  /* Fill the buffer the SPU2 isn't playing */
126  for (int i=0;i<stream_chans;i++)
127  {
128  int r;
129  r = fillbuf(stream_bufsafe, i);
130  if (r<stream_buflen) /* treat EOF and errors as the same thing atm */
131  {
132  sndStreamClose();
133  return;
134  }
135  }
136 
137  sceSdSetAddr(SD_ADDR_IRQA, stream_bufspu[0]+(stream_bufsafe*stream_buflen));
138  sceSdSetCoreAttr(SD_CORE_IRQ_ENABLE, 1);
139  }

149 int sndStreamOpen(char *file, u32 voices, u32 flags, u32 bufaddr, u32 bufsize)
150 {
151  iop_thread_t thread;
152 
153  dprintf(OUT_DEBUG, "%s\n", file);
154  if (stream_status)
155  {
156  dprintf(OUT_WARNING, "There's a stream already open\n");
157  return(-1);
158  }
159 
160  stream_voice[0] = voices&0xffff;
161  stream_voice[1] = (voices>>16)&0xffff;
162 
163  dprintf(OUT_INFO, "%s Voices %d:%d and %d:%d\n", file, (int)stream_voice[0]&1, (int)stream_voice[0]>>1, (int)stream_voice[1]&1, (int)stream_voice[1]>>1);
164 
165  if ((stream_voice[0]&1) != (stream_voice[1]&1))
166  {
167  dprintf(OUT_ERROR, "Stream voices arn't on the same core!!!!\n");
168  return(-1);
169  }
170 
171  /* Setup Semaphore */
172  if (stream_sema<0)
173  {
174  iop_sema_t sema;
175 
176  sema.attr = 0;
177  sema.option = 0;
178  sema.initial = 0;
179  sema.max = 1;
180 
181  stream_sema = CreateSema(&sema);
182  if (stream_sema<0)
183  dprintf(OUT_ERROR, "Failed to get a semaphore\n");
184  }
185 
186  stream_buflen = bufsize*16;
187  stream_flags = flags&0xffff; /* only the bottom 16bits are for user use! */
188 
189  stream_buf = AllocSysMemory(ALLOC_FIRST, stream_buflen, NULL);
190  if (stream_buf==NULL)
191  {
192  dprintf(OUT_ERROR, "malloc failed (%u bytes)\n", (unsigned int)stream_buflen);
193  return(-2);
194  }
195 
196  /* Try to open file... */
197  stream_fd = open(file, O_RDONLY);
198  if (stream_fd<0)
199  {
200  dprintf(OUT_ERROR, "open failed (%d)\n", stream_fd);
201  FreeSysMemory(stream_buf);
202  return(-3);
203  }
204 
205  stream_bufspu[0] = bufaddr;
206  stream_bufspu[1] = bufaddr + (stream_buflen*2);
207 
208  /* Setup SPU2 voice volumes.... */
209  if (stream_flags&FLAG_STEREO)
210  {
211  dprintf(OUT_INFO, "stereo...\n");
212  stream_chans = 2;
213  sceSdSetParam(stream_voice[0] | SD_VPARAM_VOLL, 0x1fff);
214  sceSdSetParam(stream_voice[0] | SD_VPARAM_VOLR, 0);
215  sceSdSetParam(stream_voice[1] | SD_VPARAM_VOLL, 0);
216  sceSdSetParam(stream_voice[1] | SD_VPARAM_VOLR, 0x1fff);
217  }
218  else
219  {
220  dprintf(OUT_INFO, "mono...\n");
221  stream_chans = 1;
222  sceSdSetParam(stream_voice[0] | SD_VPARAM_VOLL, 0x1fff);
223  sceSdSetParam(stream_voice[0] | SD_VPARAM_VOLR, 0x1fff);
224  }
225 
226  /* Setup other SPU2 voice stuff... */
227  for (int i=0;i<stream_chans;i++) /* XXX */
228  {
229  sceSdSetParam(stream_voice[i] | SD_VPARAM_PITCH, 0x1000); /* 0x1000 = normal pitch */
230  sceSdSetParam(stream_voice[i] | SD_VPARAM_ADSR1, SD_SET_ADSR1(SD_ADSR_AR_EXPi, 0, 0xf, 0xf));
231  sceSdSetParam(stream_voice[i] | SD_VPARAM_ADSR2, SD_SET_ADSR2(SD_ADSR_SR_EXPd, 127, SD_ADSR_RR_EXPd, 0));
232  }
233 
234  /* Get a thread */
235  thread.attr = TH_C;
236  thread.thread = stream_thread;
237  thread.priority = 40;
238  thread.stacksize = 0x800;
239  thread.option = 0;
240  stream_thid = CreateThread(&thread);
241  if (stream_thid<0)
242  {
243  dprintf(OUT_ERROR, "Failed to make play thread\n");
244  sndStreamClose();
245  return(-1);
246  }
247 
248  /* Start the thread (it'll wait patiently until it gets an interrupt) */
249  StartThread(stream_thid, NULL);
250 
251  /* Get some interrupts going! */
252  sceSdSetCoreAttr(SD_CORE_IRQ_ENABLE, 0); /* disable pesky interrupts for a minute */
253  sceSdSetSpu2IntrHandler(stream_handler, (void*)1);
254 
255  stream_status = STATUS_OPEN;
256 
257  stream_cur = 0;
258  if (sndStreamSetPosition(0)<0)
259  {
260  sndStreamClose();
261  return(-1);
262  }
263 
264  dprintf(OUT_INFO, "Opened %s\n", file);
265 
266  return(0);
267 }

334 int sndStreamSetPosition(int block)
335 {
336  int chunk;
337  int r=0;
338  if (stream_status==STATUS_CLOSED)
339  return(-1);
340  if (stream_status==STATUS_PLAYING)
341  r = sndStreamPause();
342  if (block<0)
343  return(-2);
344 
345  /* lock block number to a chunk */
346  chunk = (stream_buflen/16)*stream_chans;
347  dprintf(OUT_DEBUG, "chunk = %d\n", chunk);
348  block = (block/chunk)*chunk;
349 
350  stream_cur = block;
351  lseek(stream_fd, block*16*stream_chans, SEEK_SET);
352 
353  stream_bufid = 0;
354 
355  sceSdSetAddr(SD_ADDR_IRQA, stream_bufspu[0]+stream_buflen);
356 
357  for (int i=0;i<2;i++)
358  for (int c=0;c<stream_chans;c++)
359  if (fillbuf(i, c)<=0)
360  {
361  dprintf(OUT_ERROR, "Hit EOF or error on buffer fill %d\n", i);
362  sndStreamClose();
363  return(-1);
364  }
365 
366  for (int c=0;c<stream_chans;c++)
367  sceSdSetAddr(stream_voice[c] | SD_VADDR_SSA, stream_bufspu[c]);
368 
369  /* Restart playing if we were playing before */
370  if (r)
371  sndStreamPlay();
372 
373  dprintf(OUT_INFO, "Position %d!\n", stream_cur);
374 
375  return(block);
376 }
