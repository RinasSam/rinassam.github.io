112 int iLinkResetHW(void){
113  ShutDownDMAChannels();
114 
115  /* Turn off, and then turn back on the LINK and PHY. If this is not done, the iLink hardware might not function correctly on some consoles. :( */
116  ILINKRegisterBase->UnknownRegister7C=0x40; /* Shut down. */
117  DelayThread(400000);
118  ILINKRegisterBase->UnknownRegister7C=0;
119 
120  /* If the LINK is off, switch it on. */
121  if(!(ILINKRegisterBase->ctrl2&iLink_CTRL2_LPSEn)) ILINKRegisterBase->ctrl2=iLink_CTRL2_LPSEn;
122 
123  /* Wait for the clock to stabilize. */
124  while(!(ILINKRegisterBase->ctrl2&iLink_CTRL2_SOK)) DelayThread(50);
125 
126  return 0;
127 }
