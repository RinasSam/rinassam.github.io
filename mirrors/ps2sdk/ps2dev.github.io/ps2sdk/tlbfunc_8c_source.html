143 static int InitTLB32MB(void)
144 {
145  unsigned int i, NumTlbEntries, value, TlbEndIndex;
146  const struct TLBEntry *TLBEntry;
147 
148  kprintf("# TLB spad=0 kernel=1:%d default=%d:%d extended=%d:%d\n", TLBInfo.NumKernelTLBEntries - 1, TLBInfo.NumKernelTLBEntries, TLBInfo.NumKernelTLBEntries + TLBInfo.NumDefaultTLBEntries - 1, TLBInfo.NumKernelTLBEntries + TLBInfo.NumDefaultTLBEntries, TLBInfo.NumKernelTLBEntries + TLBInfo.NumDefaultTLBEntries + TLBInfo.NumExtendedTLBEntries - 1);
149 
150  __asm volatile("mtc0 $zero, $6\n"
151  "sync.p\n");
152 
153  if (TLBInfo.NumKernelTLBEntries >= 0x31) {
154  kprintf("# TLB over flow (1)");
155  Exit(1);
156  }
157 
158  for (i = 0, TLBEntry = TLBInfo.kernelTLB; i < TLBInfo.NumKernelTLBEntries; i++, TLBEntry++) {
159  _SetTLBEntry(i, TLBEntry->PageMask, TLBEntry->EntryHi, TLBEntry->EntryLo0, TLBEntry->EntryLo1);
160  }
161 
162  if (TLBInfo.NumDefaultTLBEntries + i >= 0x31) {
163  kprintf("# TLB over flow (2)");
164  Exit(1);
165  }
166 
167  for (TLBEntry = TLBInfo.defaultTLB, TlbEndIndex = TLBInfo.NumDefaultTLBEntries + i; i < TlbEndIndex; i++, TLBEntry++) {
168  _SetTLBEntry(i, TLBEntry->PageMask, TLBEntry->EntryHi, TLBEntry->EntryLo0, TLBEntry->EntryLo1);
169  }
170 
171  TLBInfo.NumWiredEntries = NumTlbEntries = i;
172  __asm volatile("mtc0 %0, $6\n"
173  "sync.p\n" ::"r"(NumTlbEntries));
174 
175  if (TLBInfo.NumExtendedTLBEntries > 0) {
176  if (TLBInfo.NumExtendedTLBEntries + i >= 0x31) {
177  kprintf("# TLB over flow (3)");
178  Exit(1);
179  }
180 
181  for (TLBEntry = TLBInfo.extendedTLB, TlbEndIndex = TLBInfo.NumExtendedTLBEntries + i; i < TlbEndIndex; i++, TLBEntry++, NumTlbEntries++) {
182  _SetTLBEntry(i, TLBEntry->PageMask, TLBEntry->EntryHi, TLBEntry->EntryLo0, TLBEntry->EntryLo1);
183  }
184  }
185 
186  for (value = 0xE0000000 + (NumTlbEntries << 13); i < 0x30; i++, value += 0x2000) {
187  _SetTLBEntry(i, 0, value, 0, 0);
188  }
189 
190  return NumTlbEntries;
191 }
