107 static void * alloc_sbrk(size_t increment)
108 {
109  u8 *mp, *ret = (void *)-1;
110 
111  if (increment == 0)
112  return _heap_ptr;
113 
114  /* If the area we want to allocated is past the end of our heap, we have a problem. */
115  mp = _heap_ptr + increment;
116  if (mp <= heap_end) {
117  ret = _heap_ptr;
118  _heap_ptr = mp;
119  }
120 
121  return ret;

126 {
127  heap_mem_header_t *prev_mem = head;
128  u32 prev_top, next_bot;
129 
130  while (prev_mem != NULL) {
131  if (prev_mem->next != NULL) {
132  prev_top = (u32)prev_mem->ptr + prev_mem->size;
133  next_bot = (u32)prev_mem->next - prev_top;
134  if (next_bot >= size)

136  }

144 void * malloc(size_t size)
145 {
146  void *ptr = NULL, *mem_ptr;
147  heap_mem_header_t *new_mem, *prev_mem;
148  size_t mem_sz, heap_align_bytes;
149 
150  mem_sz = size + sizeof(heap_mem_header_t);
151 
152  if ((mem_sz & (DEFAULT_ALIGNMENT - 1)) != 0)
153  mem_sz = ALIGN(mem_sz, DEFAULT_ALIGNMENT);
154 
155  alloc_lock();
156 
157  /* If we don't have any allocated blocks, reserve the first block from
158  the OS and initialize __alloc_heap_tail. */
159  if (__alloc_heap_head == NULL) {
160  /* Align the bottom of the heap to our default alignment. */
161  if (__alloc_heap_base == NULL) {
162  heap_align_bytes = (u32)alloc_sbrk(0) & (DEFAULT_ALIGNMENT - 1);
163  alloc_sbrk(heap_align_bytes);
164  __alloc_heap_base = alloc_sbrk(0);
165  }
166 
167  /* Allocate the physical heap and setup the head block. */
168  if ((mem_ptr = alloc_sbrk(mem_sz)) == (void *)-1)
169  return ptr; /* NULL */
170 
171  ptr = (void *)((u32)mem_ptr + sizeof(heap_mem_header_t));
172 
173  __alloc_heap_head = (heap_mem_header_t *)mem_ptr;
174  __alloc_heap_head->ptr = ptr;
175  __alloc_heap_head->size = mem_sz - sizeof(heap_mem_header_t);
176  __alloc_heap_head->prev = NULL;
177  __alloc_heap_head->next = NULL;
178 
179  __alloc_heap_tail = __alloc_heap_head;
180 
181  alloc_unlock();
182  return ptr;
183  }
184 
185  /* Check to see if there's free space at the bottom of the heap. */
186  if ((__alloc_heap_base + mem_sz) < (void *)__alloc_heap_head) {
187  new_mem = (heap_mem_header_t *)__alloc_heap_base;
188  ptr = (void *)((u32)new_mem + sizeof(heap_mem_header_t));
189 
190  new_mem->ptr = ptr;
191  new_mem->size = mem_sz - sizeof(heap_mem_header_t);
192  new_mem->prev = NULL;
193  new_mem->next = __alloc_heap_head;
194  new_mem->next->prev = new_mem;
195  __alloc_heap_head = new_mem;
196 
197  alloc_unlock();
198  return ptr;
199  }
200 
201  /* See if we can allocate the block without extending the heap. */
202  prev_mem = _heap_mem_fit(__alloc_heap_head, mem_sz);
203  if (prev_mem != NULL) {
204  new_mem = (heap_mem_header_t *)((u32)prev_mem->ptr + prev_mem->size);
205  ptr = (void *)((u32)new_mem + sizeof(heap_mem_header_t));
206 
207  new_mem->ptr = ptr;
208  new_mem->size = mem_sz - sizeof(heap_mem_header_t);
209  new_mem->prev = prev_mem;
210  new_mem->next = prev_mem->next;
211  new_mem->next->prev = new_mem;
212  prev_mem->next = new_mem;
213 
214  alloc_unlock();
215  return ptr;
216  }
217 
218  /* Extend the heap, but make certain the block is inserted in
219  order. */
220  if ((mem_ptr = alloc_sbrk(mem_sz)) == (void *)-1) {
221  alloc_unlock();
222  return ptr; /* NULL */
223  }
224 
225  ptr = (void *)((u32)mem_ptr + sizeof(heap_mem_header_t));
226 
227  new_mem = (heap_mem_header_t *)mem_ptr;
228  new_mem->ptr = ptr;
229  new_mem->size = mem_sz - sizeof(heap_mem_header_t);
230  new_mem->prev = __alloc_heap_tail;
231  new_mem->next = NULL;
232 
233  __alloc_heap_tail->next = new_mem;
234  __alloc_heap_tail = new_mem;
235 
236  alloc_unlock();
237  return ptr;
238 }

240 void * realloc(void *ptr, size_t size)
241 {
242  heap_mem_header_t *prev_mem;
243  void *new_ptr = NULL;
244 
245  if (!size && ptr != NULL) {
246  free(ptr);
247  return new_ptr;
248  }
249 
250  if (ptr == NULL)
251  return malloc(size);
252 
253  if ((size & (DEFAULT_ALIGNMENT - 1)) != 0)
254  size = ALIGN(size, DEFAULT_ALIGNMENT);
255 
256  alloc_lock();
257  prev_mem = (heap_mem_header_t *)((u32)ptr - sizeof(heap_mem_header_t));
258 
259 
260  /* Don't do anything if asked for same sized block. */
261  /* If the new size is shorter, let's just shorten the block. */
262  if (prev_mem->size >= size) {
263  /* However, if this is the last block, we have to shrink the heap. */
264  if (!prev_mem->next)
265  alloc_sbrk(ptr + size - alloc_sbrk(0));
266  prev_mem->size = size;
267 
268  alloc_unlock();
269  return ptr;
270  }
271 
272 
273  /* We are asked for a larger block of memory. */
274 
275  /* Are we the last memory block ? */
276  if (!prev_mem->next) {
277  /* Yes, let's just extend the heap then. */
278  if (alloc_sbrk(size - prev_mem->size) == (void*) -1)
279  return NULL;
280  prev_mem->size = size;
281 
282  alloc_unlock();
283  return ptr;
284  }
285 
286  /* Is the next block far enough so we can extend the current block ? */
287  if ((prev_mem->next->ptr - ptr) > size) {
288  prev_mem->size = size;
289 
290  alloc_unlock();
291  return ptr;
292  }
293 
294  alloc_unlock();
295 
296  /* We got out of luck, let's allocate a new block of memory. */
297  if ((new_ptr = malloc(size)) == NULL)
298  return new_ptr;
299 
300  /* New block is larger, we only copy the old data. */
301  memcpy(new_ptr, ptr, prev_mem->size);
302 
303  free(ptr);
304  return new_ptr;
305 }

319 void * memalign(size_t align, size_t size)
320 {
321  heap_mem_header_t new_mem;
322  heap_mem_header_t *cur_mem;
323  heap_mem_header_t *old_mem;
324  void *ptr = NULL;
325 
326  if (align <= DEFAULT_ALIGNMENT)
327  return malloc(size);
328 
329  /* Allocate with extra alignment bytes just in case it isn't aligned
330  properly by malloc. */
331  if ((ptr = malloc(size + align)) == NULL)
332  return ptr; /* NULL */
333 
334  /* If malloc returned it aligned for us we're fine. */
335  if (((u32)ptr & (align - 1)) == 0)
336  return ptr;
337 
338  alloc_lock();
339  cur_mem = (heap_mem_header_t *)((u32)ptr - sizeof(heap_mem_header_t));
340  cur_mem->size -= align;
341 
342  /* Otherwise, align the pointer and fixup our hearder accordingly. */
343  ptr = (void *)ALIGN((u32)ptr, align);
344 
345  old_mem = cur_mem;
346 
347  /* Copy the heap_mem_header_t locally, before repositioning (to make
348  sure we don't overwrite ourselves. */
349  memcpy(&new_mem, cur_mem, sizeof(heap_mem_header_t));
350  cur_mem = (heap_mem_header_t *)((u32)ptr - sizeof(heap_mem_header_t));
351  memcpy(cur_mem, &new_mem, sizeof(heap_mem_header_t));
352 
353  if (cur_mem->prev)
354  cur_mem->prev->next = cur_mem;
355  if (cur_mem->next)
356  cur_mem->next->prev = cur_mem;
357 
358  if (__alloc_heap_head == old_mem)
359  __alloc_heap_head = cur_mem;
360 
361  if (__alloc_heap_tail == old_mem)
362  __alloc_heap_tail = cur_mem;
363 
364  cur_mem->ptr = ptr;
365 
366  alloc_unlock();
367  return ptr;
368 }

370 void free(void *ptr)
371 {
372  heap_mem_header_t *cur;
373  void *heap_top;
374  size_t size;
375 
376  if (!ptr)
377  return;
378 
379  alloc_lock();
380 
381  if (!__alloc_heap_head) {
382  alloc_unlock();
383  return;
384  }
385 
386 
387  /* Freeing the head pointer is a special case. */
388  if (ptr == __alloc_heap_head->ptr) {
389  size = __alloc_heap_head->size +
390  (size_t)(__alloc_heap_head->ptr - (void *)__alloc_heap_head);
391 
392  __alloc_heap_head = __alloc_heap_head->next;
393 
394  if (__alloc_heap_head != NULL) {
395  __alloc_heap_head->prev = NULL;
396  } else {
397  __alloc_heap_tail = NULL;
398 
399  alloc_sbrk(-size);
400  }
401 
402  alloc_unlock();
403  return;
404  }
405 
406  cur = __alloc_heap_head;
407  while (ptr != cur->ptr) {
408  /* ptr isn't in our list */
409  if (cur->next == NULL) {
410  alloc_unlock();
411  return;
412  }
413 
414  cur = cur->next;
415  }
416 
417  /* Deallocate the block. */
418  if (cur->next != NULL) {
419  cur->next->prev = cur->prev;
420  } else {
421  /* If this block was the last one in the list, shrink the heap. */
422  __alloc_heap_tail = cur->prev;
423 
424  /* We need to free (heap top) - (prev->ptr + prev->size), or else
425  we'll end up with an unallocatable block of heap. */
426  heap_top = alloc_sbrk(0);
427  size = (u32)heap_top - (u32)(cur->prev->ptr + cur->prev->size);
428  alloc_sbrk(-size);
429  }
430 
431  cur->prev->next = cur->next;
432 
433  alloc_unlock();
434 }
