165 // is what XMTAPMAN does, so we have to do the same to be compatible.
166 s32 InitRpcServers()
167 {
168  iop_thread_t rpc_thread;
169 
170  // mtapPortOpen RPC Server
171  rpc_thread.attr = TH_C;
172  rpc_thread.thread = RpcThreadPortOpen;
173  rpc_thread.stacksize = 0x2000;
174  rpc_thread.priority = 32;
175 
176  threadid_rpc1 = CreateThread(&rpc_thread);
177 
178  if(threadid_rpc1 == 0) return 0;
179 
180  StartThread(threadid_rpc1, 0);
181 
182  // mtapPortClose RPC Server
183  rpc_thread.thread = RpcThreadPortClose;
184 
185  threadid_rpc2 = CreateThread(&rpc_thread);
186 
187  if(threadid_rpc2 == 0) return 0;
188 
189  StartThread(threadid_rpc2, 0);
190 
191  // mtapGetConnection RPC Server
192  rpc_thread.thread = RpcThreadGetConnection;
193 
194  threadid_rpc3 = CreateThread(&rpc_thread);
195 
196  if(threadid_rpc3 == 0) return 0;
197 
198  StartThread(threadid_rpc3, 0);
199 
200  // mtapGetSlotNumber RPC Server
201  rpc_thread.thread = rpc_thread_get_slot_number;
202 
203  threadid_rpc4 = CreateThread(&rpc_thread);
