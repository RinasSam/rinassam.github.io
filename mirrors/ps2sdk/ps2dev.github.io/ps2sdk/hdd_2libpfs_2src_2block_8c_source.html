26 {

40 // Sets 'blockpos' to point to the next block segment for the inode (and moves onto the
41 // next block descriptor inode if necessary)
42 int pfsBlockSeekNextSegment(pfs_cache_t *clink, pfs_blockpos_t *blockpos)
43 {
44  pfs_cache_t *nextSegment;
45  int result=0;
46 
47  if (blockpos->byte_offset)
48  {
49  PFS_PRINTF(PFS_DRV_NAME": Panic: This is a bug!\n");
50  return -1;
51  }
52 
53  // If we're at the end of the block descriptor array for the current
54  // inode, then move onto the next inode
55  if (pfsFixIndex(blockpos->block_segment+1)==0)
56  {
57  if ((nextSegment=pfsBlockGetNextSegment(pfsCacheUsedAdd(blockpos->inode), &result)) == NULL)
58  return result;
59  pfsCacheFree(blockpos->inode);
60  blockpos->inode=nextSegment;
61  if (clink->u.inode->number_data-1 == ++blockpos->block_segment)
62  return -EINVAL;
63  }
64 
65  blockpos->block_offset=0;
66  blockpos->block_segment++;
67  return result;
68 }

70 u32 pfsBlockSyncPos(pfs_blockpos_t *blockpos, u64 size)
71 {
72  u32 i;
73 
74  i = (u32)(size / blockpos->inode->pfsMount->zsize);
75  blockpos->byte_offset += size % blockpos->inode->pfsMount->zsize;
76 
77  if (blockpos->byte_offset >= blockpos->inode->pfsMount->zsize)
78  {
79  blockpos->byte_offset -= blockpos->inode->pfsMount->zsize;
80  i++;
81  }
82  return i;
83 }
