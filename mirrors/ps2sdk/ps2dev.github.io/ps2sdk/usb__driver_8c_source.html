148 static void usb_transfer_callback(int resultCode, int bytes, void *arg)
149 {
150  int ret;
151  usb_transfer_callback_data* data = (usb_transfer_callback_data*)arg;
152 
153  data->returnCode = resultCode;
154  if(resultCode == USB_RC_OK)
155  { //Update transfer progress if successful.
156  data->remaining -= bytes;
157  data->buffer += bytes;
158  }
159 
160  if((resultCode == USB_RC_OK) && (data->remaining > 0))
161  { //OK to continue.
162  ret = perform_bulk_transfer(data);
163  if (ret != USB_RC_OK)
164  {
165  data->returnCode = ret;
166  SignalSema(data->sema);
167  }
168  }
169  else
170  {
171  SignalSema(data->sema);
172  }
173 }

314 static int usb_bulk_manage_status(mass_dev* dev, unsigned int tag) {
315  int ret;
316  csw_packet csw;
317 
318  //XPRINTF("USBHDFSD: usb_bulk_manage_status 1 ...\n");
319  ret = usb_bulk_status(dev, &csw, tag); /* Attempt to read CSW from bulk in endpoint */
320  if (ret != USB_RC_OK) { /* STALL bulk in -OR- Bulk error */
321  usb_bulk_clear_halt(dev, USB_BLK_EP_IN); /* clear the stall condition for bulk in */
322 
323  printf("USBHDFSD: usb_bulk_manage_status error %d ...\n", ret);
324  ret = usb_bulk_status(dev, &csw, tag); /* Attempt to read CSW from bulk in endpoint */
325  }
326 
327  /* CSW not valid or stalled or phase error */
328  if (ret != USB_RC_OK || csw.signature != CSW_TAG || csw.tag != tag || csw.status == 2) {
329  printf("USBHDFSD: usb_bulk_manage_status call reset recovery ...\n");
330  usb_bulk_reset(dev, 3); /* Perform reset recovery */
331  }
332 
333  return((ret == USB_RC_OK && csw.signature == CSW_TAG && csw.tag == tag) ? csw.status : -1);
334 }

641 static int cbw_scsi_read_sector(mass_dev* dev, unsigned int lba, void* buffer, unsigned short int sectorCount)
642 {
643  int rcode, result;
644  static cbw_packet cbw={
645  CBW_TAG, // cbw.signature
646  -TAG_READ, // cbw.tag
647  0, // cbw.dataTransferLength
648  0x80, // cbw.flags
649  0, // cbw.lun
650  12, // cbw.comLength
651 
652  /* scsi command packet */
653  {
654  0x28, // Operation code
655  0, // LUN/DPO/FUA/Reserved/Reldr
656  0, // lba 1 (MSB)
657  0, // lba 2
658  0, // lba 3
659  0, // lba 4 (LSB)
660  0, // reserved
661  0, // Transfer length MSB
662  0, // Transfer length LSB
663  0, // reserved
664  0, // reserved
665  0, // reserved
666  }
667  };
668  XPRINTF("USBHDFSD: cbw_scsi_read_sector - 0x%08x %p 0x%04x\n", lba, buffer, sectorCount);
669 
670  cbw.dataTransferLength = dev->sectorSize * sectorCount;
671 
672  //scsi command packet
673  cbw.comData[2] = (lba & 0xFF000000) >> 24; //lba 1 (MSB)
674  cbw.comData[3] = (lba & 0xFF0000) >> 16; //lba 2
675  cbw.comData[4] = (lba & 0xFF00) >> 8; //lba 3
676  cbw.comData[5] = (lba & 0xFF); //lba 4 (LSB)
677  cbw.comData[7] = (sectorCount & 0xFF00) >> 8; //Transfer length MSB
678  cbw.comData[8] = (sectorCount & 0xFF); //Transfer length LSB
679 
680  result = -EIO;
681  if(usb_bulk_command(dev, &cbw) == USB_RC_OK){
682  rcode = usb_bulk_transfer(dev, USB_BLK_EP_IN, buffer, dev->sectorSize * sectorCount);
683  result = usb_bulk_manage_status(dev, -TAG_READ);
684

789 /* test that endpoint is bulk endpoint and if so, update device info */
790 static void usb_bulk_probeEndpoint(int devId, mass_dev* dev, UsbEndpointDescriptor* endpoint) {
791  if (endpoint->bmAttributes == USB_ENDPOINT_XFER_BULK) {
792  /* out transfer */
793  if ((endpoint->bEndpointAddress & USB_ENDPOINT_DIR_MASK) == USB_DIR_OUT && dev->bulkEpO < 0) {
794  /* When sceUsbdOpenPipe() is used to work around the hardware errata that occurs when an unaligned memory address is specified,
795  some USB devices become incompatible. Hence it is preferable to do alignment correction in software instead. */
796  dev->bulkEpO = sceUsbdOpenPipeAligned(devId, endpoint);
797  XPRINTF("USBHDFSD: register Output endpoint id =%i addr=%02X packetSize=%i\n", dev->bulkEpO, endpoint->bEndpointAddress, (unsigned short int)endpoint->wMaxPacketSizeHB<<8 | endpoint->wMaxPacketSizeLB);
798  } else
799  /* in transfer */
800  if ((endpoint->bEndpointAddress & USB_ENDPOINT_DIR_MASK) == USB_DIR_IN && dev->bulkEpI < 0) {
801  /* Open this pipe with sceUsbdOpenPipe, to allow unaligned addresses to be used.
802  According to the Sony documentation and the USBD code,
803  there is always an alignment check if the pipe is opened with the sceUsbdOpenPipeAligned(),
804  even when there is never any correction for the bulk out pipe. */
805  dev->bulkEpI = sceUsbdOpenPipe(devId, endpoint);
806  XPRINTF("USBHDFSD: register Input endpoint id =%i addr=%02X packetSize=%i\n", dev->bulkEpI, endpoint->bEndpointAddress, (unsigned short int)endpoint->wMaxPacketSizeHB<<8 | endpoint->wMaxPacketSizeLB);
807  }
808  }
809 }
