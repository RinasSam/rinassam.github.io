56 
57  if(!(*R_LOCAL_SBUS(PS2_SBUS_REG4) & 0x80)) { *R_LOCAL_SBUS(PS2_SBUS_REG4) = 0x80; }

66  bs = ((_sif2_xfer_chunk_size + 3) / 4);

68 
69  bc = (_sif2_xfer_chunk_size + ((bs * 4) - 1)) / (bs * 4);
