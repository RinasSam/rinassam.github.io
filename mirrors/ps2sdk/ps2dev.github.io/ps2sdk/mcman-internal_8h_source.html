196 typedef struct { // size = 384
197  char magic[28]; // Superblock magic, on PS2 MC : "Sony PS2 Memory Card Format "
198  u8 version[12]; // Version number of the format used, 1.2 indicates full support for bad_block_list
199  s16 pagesize; // size in bytes of a memory card page
200  u16 pages_per_cluster; // number of pages in a cluster
201  u16 blocksize; // number of pages in an erase block
202  u16 unused; // unused
203  u32 clusters_per_card; // total size in clusters of the memory card
204  u32 alloc_offset; // Cluster offset of the first allocatable cluster. Cluster values in the FAT and directory entries are relative to this. This is the cluster immediately after the FAT
205  u32 alloc_end; // The cluster after the highest allocatable cluster. Relative to alloc_offset. Not used
206  u32 rootdir_cluster; // First cluster of the root directory. Relative to alloc_offset. Must be zero
207  u32 backup_block1; // Erase block used as a backup area during programming. Normally the the last block on the card, it may have a different value if that block was found to be bad
208  u32 backup_block2; // This block should be erased to all ones. Normally the the second last block on the card
209  u8 unused2[8];
210  u32 ifc_list[32]; // List of indirect FAT clusters. On a standard 8M card there's only one indirect FAT cluster
211  int bad_block_list[32]; // List of erase blocks that have errors and shouldn't be used
212  u8 cardtype; // Memory card type. Must be 2, indicating that this is a PS2 memory card
213  u8 cardflags; // Physical characteristics of the memory card
214  u16 unused3;
215  u32 cluster_size;
216  u32 FATentries_per_cluster;
217  u32 clusters_per_block;
218  int cardform;
219  u32 rootdir_cluster2;
220  u32 unknown1;
221  u32 unknown2;
