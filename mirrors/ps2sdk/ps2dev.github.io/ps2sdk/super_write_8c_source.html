28 // Formats a partition (main or sub) by filling with fragment pattern and setting the bitmap accordingly
29 int pfsFormatSub(pfs_block_device_t *blockDev, int fd, u32 sub, u32 reserved, u32 scale, u32 fragment)
30 {
31  pfs_cache_t *cache;
32  int i;
33  unsigned int j;
34  u32 sector, count, size, *b;
35  int result = 0;
36 
37  size = blockDev->getSize(fd, sub);
38  sector = 1 << scale;
39  count = pfsGetBitmapSizeSectors(scale, size);
40  if (reserved>=2)
41  sector+=0x2000;
42  reserved += pfsGetBitmapSizeBlocks(scale, size);
43 
44  if((cache = pfsCacheAllocClean(&result)))
45  {
46  // fill with fragment pattern
47  for (i=127; i>=0; i--)
48  cache->u.bitmap[i]=fragment;
49 
50  // set as allocated the sectors up to reserved, for the first part of the bitmap
51  // this will mark the area the bitmaps themselves occupy as used
52  for (j=0, b=cache->u.bitmap; j<reserved; j++)
53  {
54  if (j && ((j & 0x1F)==0))
55  b++;
56  *b |= 1 << (j & 0x1F);
57  }
58 
59  PFS_PRINTF(PFS_DRV_NAME": Format sub: sub = %ld, sector start = %ld, ", sub, sector);
60 
61  // set the rest of the bitmap to the fragment
62  while (count-- && ((result=blockDev->transfer(fd, cache->u.bitmap, sub, sector++, 1, 1))>=0))
63  for (i=127; i>=0; i--)
64  cache->u.bitmap[i]=fragment;
65 
66  PFS_PRINTF("sector end = %ld\n", sector - 1);
67 
68  pfsCacheFree(cache);
69  }
70  return result;
71 }

73 // Formats a partition and all sub-partitions with PFS
74 int pfsFormat(pfs_block_device_t *blockDev, int fd, int zonesize, int fragment)
75 {
76  int result, result2;
77  pfs_cache_t *clink, *cache;
78  pfs_super_block_t *sb;
79  int scale;
80  u32 i, mainsize, subnumber = blockDev->getSubNumber(fd);
81 
82  mainsize=blockDev->getSize(fd, 0);
83  if(pfsCheckZoneSize(zonesize) == 0)
84  return -EINVAL;
85 
86  scale = pfsGetScale(zonesize, 512);
87 
88  if((clink=pfsCacheAllocClean(&result)))
89  {
90  sb = clink->u.superblock;
91  memset(sb, 0, pfsMetaSize);
92  sb->magic = PFS_SUPER_MAGIC;
93  sb->version = PFS_FORMAT_VERSION;
94  sb->modver = ((PFS_MAJOR << 8) | PFS_MINOR);
95  sb->zone_size = zonesize;
96  sb->num_subs = subnumber;
97  sb->log.number = pfsGetBitmapSizeBlocks(scale, mainsize) + (0x2000 >> scale) + 1;
98  sb->log.count = 0x20000 / zonesize ? 0x20000 / zonesize : 1;
99 
100  PFS_PRINTF(PFS_DRV_NAME": Format: log.number = %ld, log.count = %d\n", sb->log.number << scale, sb->log.count);
101 
102  sb->root.count = 1;
103  sb->root.number = sb->log.number + sb->log.count;
104  if((result = pfsJournalResetThis(blockDev, fd, sb->log.number<<scale)) >= 0)
105  {
106  if((cache = pfsCacheAllocClean(&result2)))
107  {
108  pfsFillSelfAndParentDentries(cache, &sb->root, &sb->root);
109  result2 = blockDev->transfer(fd, cache->u.dentry, 0, (sb->root.number+1) << scale,
110  1 << pfsBlockSize, 1);
111  if(result2 == 0)
112  {
113  // setup root directory
114  pfsInodeFill(cache, &sb->root, 0x11FF, 0, 0);
115  cache->u.inode->data[1].subpart = 0;
116  cache->u.inode->data[1].number = sb->root.number + 1;
117  cache->u.inode->data[1].count = 1;
118  cache->u.inode->checksum = pfsInodeCheckSum(cache->u.inode);
119 
120  result2=blockDev->transfer(fd, cache->u.inode, 0, sb->root.number << scale,
121  1 << pfsBlockSize, 1);
122  }
123 
124  pfsCacheFree(cache);
125  }
126 
127  if((result = result2) >= 0)
128  {
129  for (i=0; i < subnumber+1; i++)
130  if((result=pfsFormatSub(blockDev, fd, i, i ? 1 : (0x2000 >> scale) +
131  sb->log.count + 3, scale, fragment))<0)
132  break;
133 
134  if((result == 0) && ((result = blockDev->transfer(fd, sb, 0, PFS_SUPER_BACKUP_SECTOR, 1, 1))==0))
135  result = blockDev->transfer(fd, sb, 0, PFS_SUPER_SECTOR, 1, 1);
136  }
137  }
138 
139  pfsCacheFree(clink);
140  blockDev->flushCache(fd);
141  }
142  return result;
143 }

145 // Formats sub partitions which are added to the main partition after it is initially
146 // formatted with PFS
147 int pfsUpdateSuperBlock(pfs_mount_t *pfsMount, pfs_super_block_t *superblock, u32 sub)
148 {
149  u32 scale;
150  u32 i;
151  int rv;
152 
153  scale = pfsGetScale(superblock->zone_size, 512);
154 
155  for(i = superblock->num_subs + 1; i < sub + 1; i++)
156  {
157  rv = pfsFormatSub(pfsMount->blockDev, pfsMount->fd, i, 1, scale, 0);
158  if(rv < 0)
159  return rv;
160  }
161 
162  superblock->num_subs = sub;
163 
164  // Write superblock, then write backup
165  rv = pfsMount->blockDev->transfer(pfsMount->fd, superblock, 0, PFS_SUPER_SECTOR, 1, PFS_IO_MODE_WRITE);
166  if(!rv)
167  rv = pfsMount->blockDev->transfer(pfsMount->fd, superblock, 0, PFS_SUPER_BACKUP_SECTOR, 1, PFS_IO_MODE_WRITE);
168 
169  pfsMount->blockDev->flushCache(pfsMount->fd);
170 
171  return rv;
172 }

174 int pfsMountSuperBlock(pfs_mount_t *pfsMount)
175 {
176  int result;
177  pfs_cache_t *clink;
178  pfs_super_block_t *superblock;
179  u32 sub;
180  u32 i;
181 
182 
183  // Get number of sub partitions attached to the main partition
184  sub = pfsMount->blockDev->getSubNumber(pfsMount->fd);
185 
186  // Allocate a cache entry for the superblock
187  clink = pfsCacheAllocClean(&result);
188  if(!clink)
189  return result;
190 
191  superblock = clink->u.superblock;
192 
193  // Read the suprerblock from the main partition
194  result = pfsMount->blockDev->transfer(pfsMount->fd, superblock, 0, PFS_SUPER_SECTOR, 1, PFS_IO_MODE_READ);
195  if(result) goto error;
196 
197  if((superblock->magic != PFS_SUPER_MAGIC) || (superblock->version > PFS_FORMAT_VERSION))
198  {
199  PFS_PRINTF(PFS_DRV_NAME": Error: Invalid magic/version\n");
200  pfsCacheFree(clink);
201  return -EIO;
202  }
203 
204 
205 #ifdef PFS_SUPPORT_BHDD
206  if (strcmp(pfsMount->blockDev->devName, "bhdd") == 0)
207  {
208  if(!pfsCheckExtendedZoneSize(superblock->zone_size))
209  result = -EIO;
210  }
211  else
212 #endif
213  {
214  if(!pfsCheckZoneSize(superblock->zone_size))
215  result = -EIO;
216  }
217 
218  if((superblock->pfsFsckStat & PFS_FSCK_STAT_WRITE_ERROR) && (pfsMount->flags & PFS_FIO_ATTR_EXECUTABLE))
219  result = -EIO;
220 
221  if(sub < superblock->num_subs)
222  {
223  PFS_PRINTF(PFS_DRV_NAME": Error: Filesystem larger than partition\n");
224  result = -EIO;
225  }
226 
227  if(result) goto error;
228 
229  // If new subs have been added, update filesystem
230  if(superblock->num_subs < sub)
231  {
232  PFS_PRINTF(PFS_DRV_NAME": New subs added, updating filesystem..\n");
233  result = pfsUpdateSuperBlock(pfsMount, superblock, sub);
234  }
235 
236  if(result) goto error;
237 
238  pfsMount->zsize = superblock->zone_size;
239  pfsMount->sector_scale = pfsGetScale(pfsMount->zsize, 512);
240  pfsMount->inode_scale = pfsGetScale(pfsMount->zsize, pfsMetaSize);
241  pfsMount->num_subs = superblock->num_subs;
242  memcpy(&pfsMount->root_dir, &superblock->root, sizeof(pfs_blockinfo_t));
243  memcpy(&pfsMount->log, &superblock->log, sizeof(pfs_blockinfo_t));
244  memcpy(&pfsMount->current_dir, &superblock->root, sizeof(pfs_blockinfo_t));
245  pfsMount->total_zones = 0;
246 
247  // Do a journal restore (in case of un-clean unmount)
248  pfsJournalRestore(pfsMount);
249 
250  // Calculate free space and total size
251  for(i = 0; i < (pfsMount->num_subs + 1); i++)
252  {
253  int free;
254 
255  pfsMount->total_zones += pfsMount->blockDev->getSize(pfsMount->fd, i) >> pfsMount->sector_scale;
256 
257  free = pfsBitmapCalcFreeZones(pfsMount, i);
258  pfsMount->free_zone[i] = free;
259  pfsMount->zfree += free;
260  }
261 
262 error:
263  pfsCacheFree(clink);
264  return result;
265 }
