76 static mode_t io_to_posix_mode(unsigned int ps2mode)
77 {
78  mode_t posixmode = 0;
79  if (ps2mode & FIO_SO_IFREG) posixmode |= S_IFREG;
80  if (ps2mode & FIO_SO_IFDIR) posixmode |= S_IFDIR;
81  if (ps2mode & FIO_SO_IROTH) posixmode |= S_IRUSR|S_IRGRP|S_IROTH;
82  if (ps2mode & FIO_SO_IWOTH) posixmode |= S_IWUSR|S_IWGRP|S_IWOTH;
83  if (ps2mode & FIO_SO_IXOTH) posixmode |= S_IXUSR|S_IXGRP|S_IXOTH;
84  return posixmode;
85 }

160 /* Normalize a pathname by removing . and .. components, duplicated /, etc. */
161 static char* normalize_path(const char *path_name)
162 {
163  int i, j;
164  int first, next;
165  static char out[255];
166 
167  /* First copy the path into our temp buffer */
168  strcpy(out, path_name);
169  /* Then append "/" to make the rest easier */
170  strcat(out,"/");
171 
172  /* Convert "//" to "/" */
173  for(i=0; out[i+1]; i++) {
174  if(out[i]=='/' && out[i+1]=='/') {
175  for(j=i+1; out[j]; j++)
176  out[j] = out[j+1];
177  i--;
178  ;}
179  }
180 
181  /* Convert "/./" to "/" */
182  for(i=0; out[i] && out[i+1] && out[i+2]; i++) {
183  if(out[i]=='/' && out[i+1]=='.' && out[i+2]=='/') {
184  for(j=i+1; out[j]; j++)
185  out[j] = out[j+2];
186  i--;
187  }
188  }
189 
190  /* Convert "/path/../" to "/" until we can't anymore. Also convert leading
191  * "/../" to "/" */
192  first = next = 0;
193  while(1) {
194  /* If a "../" follows, remove it and the parent */
195  if(out[next+1] && out[next+1]=='.' &&
196  out[next+2] && out[next+2]=='.' &&
197  out[next+3] && out[next+3]=='/') {
198  for(j=0; out[first+j+1]; j++)
199  out[first+j+1] = out[next+j+4];
200  first = next = 0;
201  continue;
202  }
203 
204  /* Find next slash */
205  first = next;
206  for(next=first+1; out[next] && out[next] != '/'; next++)
207  continue;
208  if(!out[next]) break;
209  }
210 
211  /* Remove trailing "/" */
212  for(i=1; out[i]; i++)
213  continue;
214  if(i >= 1 && out[i-1] == '/')
215  out[i-1] = 0;
216 
217  return (char*)out;
218 }

227 int _open(const char *buf, int flags, ...) {
228  int iop_flags = 0;
229 
230  // newlib frags differ from iop flags
231  if ((flags & 3) == O_RDONLY) iop_flags |= IOP_O_RDONLY;
232  if ((flags & 3) == O_WRONLY) iop_flags |= IOP_O_WRONLY;
233  if ((flags & 3) == O_RDWR ) iop_flags |= IOP_O_RDWR;
234  if (flags & O_NONBLOCK) iop_flags |= IOP_O_NBLOCK;
235  if (flags & O_APPEND) iop_flags |= IOP_O_APPEND;
236  if (flags & O_CREAT) iop_flags |= IOP_O_CREAT;
237  if (flags & O_TRUNC) iop_flags |= IOP_O_TRUNC;
238  if (flags & O_EXCL) iop_flags |= IOP_O_EXCL;
239  //if (flags & O_???) iop_flags |= IOP_O_NOWAIT;
240 
241  char *t_fname = normalize_path(buf);
242  char b_fname[FILENAME_MAX];
243 
244  if (!strchr(buf, ':')) { // filename doesn't contain device
245  t_fname = b_fname;
246  if (buf[0] == '/' || buf[0] == '\\') { // does it contain root ?
247  char *device_end = strchr(__direct_pwd, ':');
248  if (device_end) { // yes, let's strip pwd a bit to keep device only
249  strncpy(b_fname, __direct_pwd, device_end - __direct_pwd);
250  strcpy(b_fname + (device_end - __direct_pwd), buf);
251  } else { // but pwd doesn't contain any device, let's default to host
252  strcpy(b_fname, "host:");
253  strcpy(b_fname + 5, buf);
254  }
255  } else { // otherwise, it's relative directory, let's copy pwd straight
256  int b_fname_len = strlen(__direct_pwd);
257  if (!strchr(__direct_pwd, ':')) { // check if pwd contains device name
258  strcpy(b_fname, "host:");
259  strcpy(b_fname + 5, __direct_pwd);
260  if (!(__direct_pwd[b_fname_len - 1] == '/' || __direct_pwd[b_fname_len - 1] == '\\')) { // does it has trailing slash ?
261  if(isCdromPath(b_fname)) {
262  b_fname[b_fname_len + 5] = '\\';
263  b_fname_len++;
264  } else {
265  b_fname[b_fname_len + 5] = '/';
266  b_fname_len++;
267  }
268  }
269  b_fname_len += 5;
270  strcpy(b_fname + b_fname_len, buf);
271  } else { // device name is here
272  if (b_fname_len) {
273  strcpy(b_fname, __direct_pwd);
274  if (!(b_fname[b_fname_len - 1] == '/' || b_fname[b_fname_len - 1] == '\\')) {
275  if(isCdromPath(b_fname)) {
276  b_fname[b_fname_len] = '\\';
277  b_fname_len++;
278  } else {
279  b_fname[b_fname_len] = '/';
280  b_fname_len++;
281  }
282  }
283  strcpy(b_fname + b_fname_len, buf);
284  }
285  }
286  }
287  }
288 
289  return __transform_errno(_ps2sdk_open(t_fname, iop_flags));
290 }
