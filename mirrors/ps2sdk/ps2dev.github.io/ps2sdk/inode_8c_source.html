67 int pfsInodeSync(pfs_blockpos_t *blockpos, u64 size, u32 used_segments)
68 {
69  int result=0;
70  u32 i;
71  u16 count;
72 
73  for(i=pfsBlockSyncPos(blockpos, size); i; )
74  {
75  count=blockpos->inode->u.inode->data[pfsFixIndex(blockpos->block_segment)].count;
76 
77  i+=blockpos->block_offset;
78 
79  if (i < count){
80  blockpos->block_offset=i;
81  break;
82  }
83 
84  i-=count;
85 
86  if (blockpos->block_segment + 1 == used_segments)
87  {
88  blockpos->block_offset=count;
89  if (i || blockpos->byte_offset){
90  PFS_PRINTF(PFS_DRV_NAME": panic: fp exceeds file.\n");
91  return -EINVAL;
92  }
93  }else{
94  blockpos->block_offset=0;
95  blockpos->block_segment++;
96  }
97 
98  if (pfsFixIndex(blockpos->block_segment))
99  continue;
100 
101  if ((blockpos->inode = pfsBlockGetNextSegment(blockpos->inode, &result)) == 0)
102  break;
103 
104  i++;
105  }
106 
107  return result;
108 }
