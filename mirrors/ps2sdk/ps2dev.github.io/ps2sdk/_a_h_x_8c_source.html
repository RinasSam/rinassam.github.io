90 void GenerateTriangle(char* Buffer, int Len)
91 {
92  int d2 = Len;
93  int d5 = d2 >> 2;
94  int d1 = 128/d5;
95  int d4 = -(d2 >> 1);
96  char* edi = Buffer;
97  char* esi;
98  int eax = 0;
99  int ecx;
100  for(ecx = 0; ecx < d5; ecx++) {
101  *edi++ = eax;
102  eax += d1;
103  }
104  *edi++ = 0x7f;
105  if(d5 != 1) {
106  eax = 128;
107  for(ecx = 0; ecx < d5-1; ecx++) {
108  eax -= d1;
109  *edi++ = eax;
110  }
111  }
112  esi = edi + d4;
113  for(ecx = 0; ecx < d5*2; ecx++) {
114  *edi++ = *esi++;
115  if(edi[-1] == 0x7f) edi[-1] = 0x80;
116  else edi[-1] = -edi[-1];
117  }
118 }

141 void GenerateWhiteNoise(char* Buffer, int Len)
142 {
143  unsigned int tmp, tmp2;
144  unsigned int Seed = 0x41595321;
145 
146  while(Len--) {
147  if(Seed&0x100) *Buffer = (char)(Seed&0xff);
148  else if(!(Seed&0xffff)) *Buffer = 0x80;
149  else *Buffer = 0x7f;
150 
151  Buffer++;
152  tmp = Seed; // ror 5
153  Seed >>= 5;
154  tmp <<= 31-5;
155  Seed |= tmp;
156  Seed = Seed ^ 0x9a;
157  tmp2 = tmp = Seed; // rol 2
158  Seed <<= 2;
159  tmp >>= 31-2;
160  Seed |= tmp;
161  Seed += tmp2;
162  tmp = Seed ^ tmp2; // ror 5
163  Seed >>= 3;
164  tmp <<= 31-3;
165  Seed |= tmp;
166  }
167 }

327 int AHXPlayer_LoadSongBuffer(void* Buffer, int Len)
328 {
329  int SongLength = Len;
330  int i, j, MaxTrack;
331  unsigned char* SongBuffer = (unsigned char*)Buffer;
332  unsigned char* SBPtr = &SongBuffer[14];
333  char* NamePtr;
334 
335  AHXPlayer_FreeMem();
336 
337  if(SongLength < 14 || SongLength == 65536) return 0;
338 
339  if(SongBuffer[0] != 'T' && SongBuffer[1] != 'H' && SongBuffer[2] != 'X') return 0;
340  Song.Revision = SongBuffer[3];
341  if(Song.Revision > 1) return 0;
342 
343  // Header ////////////////////////////////////////////
344  // Songname
345  NamePtr = (char*)&SongBuffer[(SongBuffer[4]<<8) | SongBuffer[5]];
346  Song.Name = (char*) malloc(strlen(NamePtr)+1);
347  strcpy(Song.Name, NamePtr); NamePtr += strlen(NamePtr)+1;
348  Song.SpeedMultiplier = ((SongBuffer[6]>>5)&3)+1;
349 
350  Song.PositionNr = ((SongBuffer[6]&0xf)<<8) | SongBuffer[7];
351  Song.Restart = (SongBuffer[8]<<8) | SongBuffer[9];
352  Song.TrackLength = SongBuffer[10];
353  Song.TrackNr = SongBuffer[11];
354  Song.InstrumentNr = SongBuffer[12];
355  Song.SubsongNr = SongBuffer[13];
356 
357  // Subsongs //////////////////////////////////////////
358  Song.Subsongs = (int*) malloc(sizeof(int)*Song.SubsongNr);
359  #ifndef COMPACT_CODE
360  printf("name length = %d\n",strlen(NamePtr)+1);
361  printf("playing: %s\n", NamePtr);
362  printf("subsongs = %d\n",Song.SubsongNr);
363  printf("Positions = %d\n",Song.PositionNr);
364  printf("Tracks = %d\n",(MaxTrack+1));
365  #endif
366  for(i = 0; i < Song.SubsongNr; i++) {
367  if(SBPtr - SongBuffer > SongLength) return 0;
368  Song.Subsongs[i] = (SBPtr[0]<<8)|SBPtr[1];
369  SBPtr += 2;
370  }
371 
372  // Position List /////////////////////////////////////
373  Positions = (struct AHXPosition*) malloc(sizeof(struct AHXPosition)*Song.PositionNr);
374  for(i = 0; i < Song.PositionNr; i++) {
375  for(j = 0; j < 4; j++) {
376  if(SBPtr - SongBuffer > SongLength) return 0;
377  Positions[i].Track[j] = *SBPtr++;
378  Positions[i].Transpose[j] = *(signed char*)SBPtr++;
379  }
380  }
381 
382  // Tracks ////////////////////////////////////////////
383  MaxTrack = Song.TrackNr;
384  Tracks = (struct AHXStep**) malloc(sizeof(struct AHXStep)*(MaxTrack+1));
385  for(i = 0; i < MaxTrack+1; i++) {
386  Tracks[i] = (struct AHXStep*) malloc(sizeof(struct AHXStep)*Song.TrackLength);
387  if((SongBuffer[6]&0x80)==0x80 && i==0) {
388  memset(Tracks[i], 0, Song.TrackLength*sizeof(struct AHXStep));
389  continue;
390  }
391  for(j = 0; j < Song.TrackLength; j++) {
392  if(SBPtr - SongBuffer > SongLength) return 0;
393  Tracks[i][j].Note = (SBPtr[0]>>2)&0x3f;
394  Tracks[i][j].Instrument = ((SBPtr[0]&0x3)<<4) | (SBPtr[1]>>4);
395  Tracks[i][j].FX = SBPtr[1]&0xf;
396  Tracks[i][j].FXParam = SBPtr[2];
397  SBPtr += 3;
398  }
399  }
400 
401  // Instruments ///////////////////////////////////////
402  Instruments = (struct AHXInstrument*) malloc(sizeof(struct AHXInstrument)*(Song.InstrumentNr+1));
403  for(i = 1; i < Song.InstrumentNr+1; i++) {
404  Instruments[i].Name = (char*) malloc(strlen(NamePtr)+1);
405  strcpy(Instruments[i].Name, NamePtr); NamePtr += strlen(NamePtr)+1;
406  if(SBPtr - SongBuffer > SongLength) return 0;
407  Instruments[i].Volume = SBPtr[0];
408  Instruments[i].FilterSpeed = ((SBPtr[1]>>3)&0x1f) | ((SBPtr[12]>>2)&0x20);
409  Instruments[i].WaveLength = SBPtr[1]&0x7;
410  Instruments[i].Envelope.aFrames = SBPtr[2];
411  Instruments[i].Envelope.aVolume = SBPtr[3];
412  Instruments[i].Envelope.dFrames = SBPtr[4]; //4
413  Instruments[i].Envelope.dVolume = SBPtr[5];
414  Instruments[i].Envelope.sFrames = SBPtr[6];
415  Instruments[i].Envelope.rFrames = SBPtr[7]; //7
416  Instruments[i].Envelope.rVolume = SBPtr[8];
417  Instruments[i].FilterLowerLimit = SBPtr[12]&0x7f;
418  Instruments[i].VibratoDelay = SBPtr[13]; //13
419  Instruments[i].HardCutReleaseFrames = (SBPtr[14]>>4)&7;
420  Instruments[i].HardCutRelease = SBPtr[14]&0x80?1:0;
421  Instruments[i].VibratoDepth = SBPtr[14]&0xf; //14
422  Instruments[i].VibratoSpeed = SBPtr[15];
423  Instruments[i].SquareLowerLimit = SBPtr[16];
424  Instruments[i].SquareUpperLimit = SBPtr[17]; //17
425  Instruments[i].SquareSpeed = SBPtr[18];
426  Instruments[i].FilterUpperLimit = SBPtr[19]&0x3f; //19
427  Instruments[i].PList.Speed = SBPtr[20];
428  Instruments[i].PList.Length= SBPtr[21];
429  SBPtr += 22;
430  Instruments[i].PList.Entries = (struct AHXPListEntry*) malloc(sizeof(struct AHXPListEntry)*Instruments[i].PList.Length);
431  for(j = 0; j < Instruments[i].PList.Length; j++) {
432  if(SBPtr - SongBuffer > SongLength) return 0;
433  Instruments[i].PList.Entries[j].FX[1] = (SBPtr[0]>>5)&7;
434  Instruments[i].PList.Entries[j].FX[0] = (SBPtr[0]>>2)&7;
435  Instruments[i].PList.Entries[j].Waveform = ((SBPtr[0]<<1)&6) | (SBPtr[1]>>7);
436  Instruments[i].PList.Entries[j].Fixed = (SBPtr[1]>>6)&1;
437  Instruments[i].PList.Entries[j].Note = SBPtr[1]&0x3f;
438  Instruments[i].PList.Entries[j].FXParam[0] = SBPtr[2];
439  Instruments[i].PList.Entries[j].FXParam[1] = SBPtr[3];
440  SBPtr += 4;
441  }
442  }
443  return Song.SubsongNr;
444 }

503 void AHXPlayer_ProcessStep(int v)
504 {
505  int Note, Instrument, FX, FXParam, SquareLower, SquareUpper, d3, d4, d6, Neue, Alte, i;
506 
507  if(!Voices[v].TrackOn) return;
508  Voices[v].VolumeSlideUp = Voices[v].VolumeSlideDown = 0;
509 
510  Note = Tracks[Positions[PosNr].Track[v]][NoteNr].Note;
511  Instrument = Tracks[Positions[PosNr].Track[v]][NoteNr].Instrument;
512  FX = Tracks[Positions[PosNr].Track[v]][NoteNr].FX;
513  FXParam = Tracks[Positions[PosNr].Track[v]][NoteNr].FXParam;
514 
515  switch(FX) {
516  case 0x0: // Position Jump HI
517  if((FXParam & 0xf) > 0 && (FXParam & 0xf) <= 9)
518  PosJump = FXParam & 0xf;
519  break;
520  case 0x5: // Volume Slide + Tone Portamento
521  case 0xa: // Volume Slide
522  Voices[v].VolumeSlideDown = FXParam & 0x0f;
523  Voices[v].VolumeSlideUp = FXParam >> 4;
524  break;
525  case 0xb: // Position Jump
526  PosJump = PosJump*100 + (FXParam & 0x0f) + (FXParam >> 4)*10;
527  PatternBreak = 1;
528  break;
529  case 0xd: // Patternbreak
530  PosJump = PosNr + 1;
531  PosJumpNote = (FXParam & 0x0f) + (FXParam >> 4)*10;
532  if(PosJumpNote > Song.TrackLength) PosJumpNote = 0;
533  PatternBreak = 1;
534  break;
535  case 0xe: // Enhanced commands
536  switch(FXParam >> 4) {
537  case 0xc: // Note Cut
538  if((FXParam & 0x0f) < Tempo) {
539  Voices[v].NoteCutWait = FXParam & 0x0f;
540  if(Voices[v].NoteCutWait) {
541  Voices[v].NoteCutOn = 1;
542  Voices[v].HardCutRelease = 0;
543  }
544  }
545  break;
546  case 0xd: // Note Delay
547  if(Voices[v].NoteDelayOn) {
548  Voices[v].NoteDelayOn = 0;
549  } else {
550  if((FXParam & 0x0f) < Tempo) {
551  Voices[v].NoteDelayWait = FXParam & 0x0f;
552  if(Voices[v].NoteDelayWait) {
553  Voices[v].NoteDelayOn = 1;
554  return;
555  }
556  }
557  }
558  break;
559  }
560  break;
561  case 0xf: // Speed
562  Tempo = FXParam;
563  break;
564  }
565  if(Instrument) {
566  Voices[v].PerfSubVolume = 0x40;
567  Voices[v].PeriodSlideSpeed = Voices[v].PeriodSlidePeriod = Voices[v].PeriodSlideLimit = 0;
568  Voices[v].ADSRVolume = 0;
569  Voices[v].Instrument = &Instruments[Instrument];
570  AHXVoice_CalcADSR(&Voices[v]);
571  //InitOnInstrument
572  Voices[v].WaveLength = Voices[v].Instrument->WaveLength;
573  Voices[v].NoteMaxVolume = Voices[v].Instrument->Volume;
574  //InitVibrato
575  Voices[v].VibratoCurrent = 0;
576  Voices[v].VibratoDelay = Voices[v].Instrument->VibratoDelay;
577  Voices[v].VibratoDepth = Voices[v].Instrument->VibratoDepth;
578  Voices[v].VibratoSpeed = Voices[v].Instrument->VibratoSpeed;
579  Voices[v].VibratoPeriod = 0;
580  //InitHardCut
581  Voices[v].HardCutRelease = Voices[v].Instrument->HardCutRelease;
582  Voices[v].HardCut = Voices[v].Instrument->HardCutReleaseFrames;
583  //InitSquare
584  Voices[v].IgnoreSquare = Voices[v].SquareSlidingIn = 0;
585  Voices[v].SquareWait = Voices[v].SquareOn = 0;
586  SquareLower = Voices[v].Instrument->SquareLowerLimit >> (5-Voices[v].WaveLength);
587  SquareUpper = Voices[v].Instrument->SquareUpperLimit >> (5-Voices[v].WaveLength);
588  if(SquareUpper < SquareLower) { int t = SquareUpper; SquareUpper = SquareLower; SquareLower = t; }
589  Voices[v].SquareUpperLimit = SquareUpper;
590  Voices[v].SquareLowerLimit = SquareLower;
591  //InitFilter
592  Voices[v].IgnoreFilter = Voices[v].FilterWait = Voices[v].FilterOn = 0;
593  Voices[v].FilterSlidingIn = 0;
594  d6 = Voices[v].Instrument->FilterSpeed;
595  d3 = Voices[v].Instrument->FilterLowerLimit;
596  d4 = Voices[v].Instrument->FilterUpperLimit;
597  if(d3 & 0x80) d6 |= 0x20;
598  if(d4 & 0x80) d6 |= 0x40;
599  Voices[v].FilterSpeed = d6;
600  d3 &= ~0x80;
601  d4 &= ~0x80;
602  if(d3 > d4) { int t = d3; d3 = d4; d4 = t; }
603  Voices[v].FilterUpperLimit = d4;
604  Voices[v].FilterLowerLimit = d3;
605  Voices[v].FilterPos = 32;
606  //Init PerfList
607  Voices[v].PerfWait = Voices[v].PerfCurrent = 0;
608  Voices[v].PerfSpeed = Voices[v].Instrument->PList.Speed;
609  Voices[v].PerfList = &Voices[v].Instrument->PList;
610  }
611  //NoInstrument
612  Voices[v].PeriodSlideOn = 0;
613 
614  switch(FX) {
615  case 0x4: // Override filter
616  break;
617  case 0x9: // Set Squarewave-Offset
618  Voices[v].SquarePos = FXParam >> (5 - Voices[v].WaveLength);
619  Voices[v].PlantSquare = 1;
620  Voices[v].IgnoreSquare = 1;
621  break;
622  case 0x5: // Tone Portamento + Volume Slide
623  case 0x3: // Tone Portamento (Period Slide Up/Down w/ Limit)
624  if(FXParam != 0) Voices[v].PeriodSlideSpeed = FXParam;
625  if(Note) {
626  Neue = PeriodTable[Note];
627  Alte = PeriodTable[Voices[v].TrackPeriod];
628  Alte -= Neue;
629  Neue = Alte + Voices[v].PeriodSlidePeriod;
630  if(Neue) Voices[v].PeriodSlideLimit = -Alte;
631  }
632  Voices[v].PeriodSlideOn = 1;
633  Voices[v].PeriodSlideWithLimit = 1;
634  goto NoNote;
635  }
636 
637  // Note anschlagen
638  if(Note) {
639  Voices[v].TrackPeriod = Note;
640  Voices[v].PlantPeriod = 1;
641  }
642 NoNote:
643  switch(FX) {
644  case 0x1: // Portamento up (Period slide down)
645  Voices[v].PeriodSlideSpeed = -FXParam;
646  Voices[v].PeriodSlideOn = 1;
647  Voices[v].PeriodSlideWithLimit = 0;
648  break;
649  case 0x2: // Portamento down (Period slide up)
650  Voices[v].PeriodSlideSpeed = FXParam;
651  Voices[v].PeriodSlideOn = 1;
652  Voices[v].PeriodSlideWithLimit = 0;
653  break;
654  case 0xc: // Volume
655  if(FXParam <= 0x40)
656  Voices[v].NoteMaxVolume = FXParam;
657  else {
658  FXParam -= 0x50;
659  if(FXParam <= 0x40)
660  for(i = 0; i < 4; i++) Voices[i].TrackMasterVolume = FXParam;
661  else {
662  FXParam -= 0xa0 - 0x50;
663  if(FXParam <= 0x40)
664  Voices[v].TrackMasterVolume = FXParam;
665  }
666  }
667  break;
668  case 0xe: // Enhanced commands
669  switch(FXParam >> 4) {
670  case 0x1: // Fineslide up (Period fineslide down)
671  Voices[v].PeriodSlidePeriod = -(FXParam & 0x0f);
672  Voices[v].PlantPeriod = 1;
673  break;
674  case 0x2: // Fineslide down (Period fineslide up)
675  Voices[v].PeriodSlidePeriod = FXParam & 0x0f;
676  Voices[v].PlantPeriod = 1;
677  break;
678  case 0x4: // Vibrato control
679  Voices[v].VibratoDepth = FXParam & 0x0f;
680  break;
681  case 0xa: // Finevolume up
682  Voices[v].NoteMaxVolume += FXParam & 0x0f;
683  if(Voices[v].NoteMaxVolume > 0x40) Voices[v].NoteMaxVolume = 0x40;
684  break;
685  case 0xb: // Finevolume down
686  Voices[v].NoteMaxVolume -= FXParam & 0x0f;
687  if(Voices[v].NoteMaxVolume < 0) Voices[v].NoteMaxVolume = 0;
688  break;
689  }
690  break;
691  }

755 void AHXPlayer_ProcessFrame(int v)
756 {

762  if(Voices[v].NoteDelayOn) {
763  if(Voices[v].NoteDelayWait <= 0) AHXPlayer_ProcessStep(v);
764  else Voices[v].NoteDelayWait--;
765  }
766  if(Voices[v].HardCut) {
767  int NextInstrument;
768  if(NoteNr+1 < Song.TrackLength) NextInstrument = Tracks[Voices[v].Track][NoteNr+1].Instrument;
769  else NextInstrument = Tracks[Voices[v].NextTrack][0].Instrument;

795  Voices[v].ADSRVolume += Voices[v].ADSR.dVolume; // Delta

812  if(d0 > 0) d2 = -d2;
813  if(d0) {
814  d3 = (d0 + d2) ^ d0;
815  if(d3 >= 0) d0 = Voices[v].PeriodSlidePeriod + d2;
816  else d0 = Voices[v].PeriodSlideLimit;
817  Voices[v].PeriodSlidePeriod = d0;
818  Voices[v].PlantPeriod = 1;
819  }
820  } else {
821  Voices[v].PeriodSlidePeriod += Voices[v].PeriodSlideSpeed;
822  Voices[v].PlantPeriod = 1;

829  Voices[v].PlantPeriod = 1;
830  Voices[v].VibratoCurrent = (Voices[v].VibratoCurrent + Voices[v].VibratoSpeed) & 0x3f;
831  } else Voices[v].VibratoDelay--;
832  }

839  Voices[v].Waveform = Voices[v].PerfList->Entries[Cur].Waveform-1;

842  }

849  Voices[v].PlantPeriod = 1;
850  Voices[v].FixedNote = Voices[v].PerfList->Entries[Cur].Fixed;
851  }
852  }

854  if(Voices[v].PerfWait) Voices[v].PerfWait--;
855  else Voices[v].PeriodPerfSlideSpeed = 0;
856  }

865  int d2 = Voices[v].SquareUpperLimit;

870  Voices[v].SquareSlidingIn = 1;
871  Voices[v].SquareSign = 1;
872  } else if(d3 >= d2) {
873  Voices[v].SquareSlidingIn = 1;
874  Voices[v].SquareSign = -1;
875  }
876  }
877  //NoSquareInit
878  if(d1 == d3 || d2 == d3) {
879  if(Voices[v].SquareSlidingIn) {
880  Voices[v].SquareSlidingIn = 0;
881  } else {
882  Voices[v].SquareSign = -Voices[v].SquareSign;
883  }
884  }

886  Voices[v].SquarePos = d3;
887  Voices[v].PlantSquare = 1;
888  Voices[v].SquareWait = Voices[v].Instrument->SquareSpeed;
889  }
890  }
891  if(Voices[v].FilterOn && --Voices[v].FilterWait <= 0) {
892  d1 = Voices[v].FilterLowerLimit;
893  d2 = Voices[v].FilterUpperLimit;
894  d3 = Voices[v].FilterPos;
895  if(Voices[v].FilterInit) {
896  Voices[v].FilterInit = 0;
897  if(d3 <= d1) {
898  Voices[v].FilterSlidingIn = 1;
899  Voices[v].FilterSign = 1;
900  } else if(d3 >= d2) {
901  Voices[v].FilterSlidingIn = 1;
902  Voices[v].FilterSign = -1;
903  }
904  }
905  //NoFilterInit
906  FMax = (Voices[v].FilterSpeed < 3)?(5-Voices[v].FilterSpeed):1;
907  for(i = 0; i < FMax; i++) {
908  if(d1 == d3 || d2 == d3) {
909  if(Voices[v].FilterSlidingIn) {
910  Voices[v].FilterSlidingIn = 0;
911  } else {
912  Voices[v].FilterSign = -Voices[v].FilterSign;
913  }
914  }
915  d3 += Voices[v].FilterSign;
916  }
917  Voices[v].FilterPos = d3;
918  Voices[v].NewWaveform = 1;
919  Voices[v].FilterWait = Voices[v].FilterSpeed - 3;
920  if(Voices[v].FilterWait < 1) Voices[v].FilterWait = 1;
921  }
922  if(Voices[v].Waveform == 3-1 || Voices[v].PlantSquare) {

924  SquarePtr = &Waves->Squares[(Voices[v].FilterPos-0x20)*(0xfc+0xfc+0x80*0x1f+0x80+0x280*3)];

927  X = 0x40 - X;

936  SquarePtr += Delta;
937  }
938  Voices[v].NewWaveform = 1;
939  Voices[v].Waveform = 3-1;
940  Voices[v].PlantSquare = 0;
941  }
942  if(Voices[v].Waveform == 4-1) Voices[v].NewWaveform = 1;
943 
944  if(Voices[v].NewWaveform) {
945  AudioSource = WaveformTab[Voices[v].Waveform];
946  if(Voices[v].Waveform != 3-1) {
947  AudioSource += (Voices[v].FilterPos-0x20)*(0xfc+0xfc+0x80*0x1f+0x80+0x280*3);

950  AudioSource += Offsets[Voices[v].WaveLength];
951  }

955  //GoOnRandom
956  WNRandom += 2239384;
957  WNRandom = ((((WNRandom >> 8) | (WNRandom << 24)) + 782323) ^ 75) - 6735;
958  }
959  Voices[v].AudioSource = AudioSource;
960  }
961  //StillHoldWaveform
962  //AudioInitPeriod
963  Voices[v].AudioPeriod = Voices[v].InstrPeriod;
964  if(!Voices[v].FixedNote) Voices[v].AudioPeriod += Voices[v].Transpose + Voices[v].TrackPeriod-1;
965  if(Voices[v].AudioPeriod > 5*12) Voices[v].AudioPeriod = 5*12;

968  if(!Voices[v].FixedNote) Voices[v].AudioPeriod += Voices[v].PeriodSlidePeriod;
969  Voices[v].AudioPeriod += Voices[v].PeriodPerfSlidePeriod + Voices[v].VibratoPeriod;
970  if(Voices[v].AudioPeriod > 0x0d60) Voices[v].AudioPeriod = 0x0d60;
971  if(Voices[v].AudioPeriod < 0x0071) Voices[v].AudioPeriod = 0x0071;
972  //AudioInitVolume

974 }
