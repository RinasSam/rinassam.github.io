94 static void PrntFunc(void *context, int chr)
95 {
96  KprArg *kpa = (KprArg *)context;
97 
98  switch (chr) {
99  case 0:
100  break;
101  case PRNT_IO_BEGIN:
102  kpa->calls++;
103  break;
104  case PRNT_IO_END:
105  break;
106  case '\n':
107  PrntFunc(context, '\r');
108  default:
109  if (kpa->prpos < kpa->bsize)
110  kpa->kpbuf[kpa->prpos++] = chr;
111  break;
112  }
113 }

138 static void KPRTTY_Thread(void *args)
139 {
140  u32 flags;
141  KprArg *kpa = (KprArg *)args;
142 
143  while (1) {
144  WaitEventFlag(kpa->eflag, 1, WEF_AND | WEF_CLEAR, &flags);
145 
146  if (kpa->prpos) {
147  write(1, kpa->kpbuf, kpa->prpos);
148  kpa->prpos = 0;
149  }
150  }

185 int _start(int argc, char **argv)
186 {

188  RegisterLibraryEntries(&_exp_udptty);
189 
190  // create the socket
191  udp_socket = lwip_socket(AF_INET, SOCK_DGRAM, IPPROTO_UDP);
192  if (udp_socket < 0)
193  return MODULE_NO_RESIDENT_END;
194 
195  close(0);
196  close(1);
197  DelDrv(DEVNAME);
198 
199  if (AddDrv(&tty_device) < 0)
200  return MODULE_NO_RESIDENT_END;
201 
202  open(DEVNAME "00:", 0x1000 | O_RDWR);
203  open(DEVNAME "00:", O_WRONLY);
204 
205  printf("UDPTTY loaded!\n");
206 
207 #ifdef KPRTTY
208  kprtty_init();
209  printf("KPRTTY enabled!\n");
210 #endif
211 
212  return MODULE_RESIDENT_END;

225 {
226  struct sockaddr_in peer;

228  peer.sin_family = AF_INET;
229  peer.sin_port = htons(18194);
230  peer.sin_addr.s_addr = inet_addr("255.255.255.255");
231 
232  lwip_sendto(udp_socket, buf, size, 0, (struct sockaddr *)&peer, sizeof(peer));

234  return 0;
