67 // UID and GID
68 /* UID and GID are fixed with constants.
69  Files (and directories) created by the system have UID and GID set to 0,
70  as they were made with an older library.
71 
72  As of release v2.3, the UID and GID were changed to 0xFFFF.
73  The HDD Browser probably changes the UID and GID to 0xFFFF with chstat,
74  although I have not verified that.
75 
76  If the UID and GID are not set to 0xFFFF, then the software may ignore the entry. */
77 #define PFS_UID 0xFFFF
78 #define PFS_GID 0xFFFF
