130 int _start(int argc, char *argv[])
131 {
132  char *filename;
133  int number;
134  int numBuf = 8;
135  int reqBuf;
136  int size, ret;
137 
138  PFS_PRINTF(PFS_DRV_NAME" Playstation Filesystem Driver v%d.%d\nps2fs: (c) 2003 Sjeep, Vector and Florin Sasu\n", PFS_MAJOR, PFS_MINOR);
139 
140  // Get filename of IRX
141  filename = strrchr(argv[0], '/');
142  if(filename++)
143  pfsFilename = filename;
144  else
145  pfsFilename = argv[0];
146 
147  argc--;
148  argv++;
149 
150  // Parse arguments
151  while(argc > 0)
152  {
153  if(argv[0][0] != '-')
154  break;
155 
156  if(!strcmp(argv[0], "-m"))
157  {
158  if(--argc <= 0)
159  return printPfsArgError();
160  argv++;
161 
162  number = strtol(argv[0], 0, 10);
163 
164  if(number <= 32)
165  pfsConfig.maxMount = number;
166  }
167  else if(!strcmp(argv[0], "-o"))
168  {
169  if(--argc <= 0)
170  return printPfsArgError();
171  argv++;
172 
173  number = strtol(argv[0], NULL, 10);
174 
175  if(number <= 32)
176  pfsConfig.maxOpen = number;
177  }
178  else if(!strcmp(argv[0], "-n"))
179  {
180  if(--argc <= 0)
181  return printPfsArgError();
182  argv++;
183 
184  number = strtol(argv[0], NULL, 10);
185 
186  if(number > numBuf)
187  numBuf = number;
188 
189  if(numBuf > 127) {
190  PFS_PRINTF(PFS_DRV_NAME" ERROR: Number of buffers is larger than 127!\n");
191  return -EINVAL;
192  }
193  }
194  else
195  return printPfsArgError();
196 
197  argc--;
198  argv++;
199  }
200 
201  PFS_PRINTF(PFS_DRV_NAME" Max mount: %ld, Max open: %ld, Number of buffers: %d\n", pfsConfig.maxMount,
202  pfsConfig.maxOpen, numBuf);
203 
204  // Do we have enough buffers ?
205  reqBuf = (pfsConfig.maxOpen * 2) + 8;
206  if(numBuf < reqBuf)
207  PFS_PRINTF(PFS_DRV_NAME" Warning: %d buffers may be needed, but only %d buffers are allocated\n", reqBuf, numBuf);
208 
209  if(allocateMountBuffer(pfsConfig.maxMount) < 0)
210  return MODULE_NO_RESIDENT_END;
211 
212  // Allocate and zero memory for file slots
213  size = pfsConfig.maxOpen * sizeof(pfs_file_slot_t);
214  pfsFileSlots = pfsAllocMem(size);
215  ret = (pfsFileSlots == NULL) ? -ENOMEM : 0;
216  if(ret != 0)
217  { //Official PFS module does not print an error message here.
218  PFS_PRINTF(PFS_DRV_NAME" Error: Failed to allocate memory for file descriptors!\n");
219  return MODULE_NO_RESIDENT_END;
220  }
221 
222  memset(pfsFileSlots, 0, size);
223 
224  if(pfsCacheInit(numBuf, pfsMetaSize) < 0)
225  return MODULE_NO_RESIDENT_END;
226 
227  DelDrv("pfs");
228  if(AddDrv(&pfsFioDev) == 0) {
229 #if defined(PFS_XOSD_VER)
230  PFS_PRINTF(PFS_DRV_NAME" version %04x driver start. This is OSD LBA48 VERSION !!!!!!!!!!!\n", IRX_VER(PFS_MAJOR, PFS_MINOR));
231 #elif defined(PFS_OSD_VER)
232  PFS_PRINTF(PFS_DRV_NAME" version %04x driver start. This is OSD version!\n", IRX_VER(PFS_MAJOR, PFS_MINOR));
233 #else
234  PFS_PRINTF(PFS_DRV_NAME" version %04x driver start.\n", IRX_VER(PFS_MAJOR, PFS_MINOR));
235 #endif
236  return MODULE_RESIDENT_END;
237  }
238 
239  return MODULE_NO_RESIDENT_END;
240 }
