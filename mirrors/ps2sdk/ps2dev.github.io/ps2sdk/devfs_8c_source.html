177  dirent->stat.mode |= FIO_S_IWUSR;
178  }

376 
377  //printf("devfs_open file=%p name=%s mode=%d\n", file, name, mode);
378  if(name == NULL)
379  {
380  M_PRINTF("open: Name is NULL\n");
381  return -1;
382  }
383 
384  if((name[0] == '\\') || (name[0] == '/'))
385  {
386  fn_offset = 1;
387  }
388 
389  dev = devfs_find_devicename(name + fn_offset);
390 
391  if(dev != NULL)
392  {
393  char *endp;
394  int subdev;
395  int name_len;
396 
397  name_len = strlen(dev->node.name);
398 
399  if(name_len == strlen(name)) /* If this is has not got a subnumber */
400  {
401  M_PRINTF("open: No subdevice number in filename %s\n", name);
402  return -1;
403  }
404 
405  subdev = strtoul(&name[name_len + fn_offset], &endp, 10);
406  if(((subdev == 0) && (name[name_len + fn_offset] != '0'))
407  || (subdev >= DEVFS_MAX_SUBDEVS))
408  /* Invalid number */
409  {
410  M_PRINTF("open: Invalid subdev number %d\n", subdev);
411  return -1;
412  }
413 
414  if(*endp)
415  /* Extra charactes after filename */
416  {
417  M_PRINTF("open: Invalid filename\n");
418  return -1;
419  }
420 
421  if(!dev->subdevs[subdev].valid)
422  /* No subdev */
423  {
424  M_PRINTF("open: No subdev registered\n");
425  return -1;
426  }
427 
428  if((dev->subdevs[subdev].mode & DEVFS_MODE_EX)
429  && (dev->subdevs[subdev].open_refcount > 0))
430  /* Already opened in exclusive mode */
431  {
432  M_PRINTF("open: Exclusive subdevice already opened\n");
433  return -1;
434  }
435 
436  if(dev->subdevs[subdev].open_refcount == MAX_OPENFILES)
437  {
438  M_PRINTF("open: Reached open file limit for sub device\n");
439  }
440 
441  /* Tried to open read but not allowed */
442  if(((mode & O_RDONLY) || ((mode & O_RDWR) == O_RDWR))
443  && !(dev->subdevs[subdev].mode & DEVFS_MODE_R))
444  {
445  M_PRINTF("open: Read mode requested but not permitted\n");
446  return -1;
447  }
448 
449  if(((mode & O_WRONLY) || ((mode & O_RDWR) == O_RDWR))
450  && !(dev->subdevs[subdev].mode & DEVFS_MODE_W))
451  {
452  M_PRINTF("open: Write mode requested but not permitted\n");
453  return -1;
454  }
455 
456  file->privdata = AllocSysMemory(ALLOC_FIRST, sizeof(ioman_data_t), NULL);
457  if(file->privdata == NULL)
458  {
459  M_PRINTF("open: Allocation failure\n");
460  return -1;
461  }
462 
463  ((ioman_data_t *) file->privdata)->hDev = dev->hDev;
464  ((ioman_data_t *) file->privdata)->subdev = subdev;
465  ((ioman_data_t *) file->privdata)->loc.loc64 = 0;
466  ((ioman_data_t *) file->privdata)->dev = dev;
467  ((ioman_data_t *) file->privdata)->mode = mode;
468 
469  dev->subdevs[subdev].open_refcount++;
470  loop = 0;
471  while((loop < MAX_OPENFILES) && (dev->subdevs[subdev].open_files[loop]))
472  {
473  loop++;
474  }
475  if(loop == MAX_OPENFILES)
476  {
477  FreeSysMemory(file->privdata);
478  M_PRINTF("open: Inconsistency between number of open files and available slot\n");
479  return -1;
480  }
481 
482  dev->subdevs[subdev].open_files[loop] = file->privdata;
483  dev->open_refcount++;
484  M_PRINTF("open: Opened device %s subdev %d\n", dev->node.name, subdev);
485  }
486  else
487  {
488  M_PRINTF("open: Couldn't find the device\n");
489  return -1;
490  }
491 
492  return 0;
