182 void main_thread(void *unused)
183 {
184  u32 resbits[4];
185 
186  while (1) {
187  #ifdef SIO2LOG
188  log_flush(0);
189  #endif
190  WaitEventFlag(event_flag, EF_PAD_TRANSFER_INIT |
191  EF_MC_TRANSFER_INIT | EF_MTAP_TRANSFER_INIT
192 #ifdef SIO2MAN_V2
193  | EF_RM_TRANSFER_INIT | EF_UNK_TRANSFER_INIT
194 #endif
195  , 1, resbits);
196 
197  if (resbits[0] & EF_PAD_TRANSFER_INIT) {
198  ClearEventFlag(event_flag, ~EF_PAD_TRANSFER_INIT);
199  SetEventFlag(event_flag, EF_PAD_TRANSFER_READY);
200 #ifdef SIO2LOG
201  log_default(LOG_PAD_READY);
202 #endif
203  } else if (resbits[0] & EF_MC_TRANSFER_INIT) {
204  ClearEventFlag(event_flag, ~EF_MC_TRANSFER_INIT);
205  SetEventFlag(event_flag, EF_MC_TRANSFER_READY);
206 #ifdef SIO2LOG
207  log_default(LOG_MC_READY);
208 #endif
209  } else if (resbits[0] & EF_MTAP_TRANSFER_INIT) {
210  ClearEventFlag(event_flag, ~EF_MTAP_TRANSFER_INIT);
211  SetEventFlag(event_flag, EF_MTAP_TRANSFER_READY);
212 #ifdef SIO2LOG
213  log_default(LOG_MTAP_READY);
214 #endif
215 #ifdef SIO2MAN_V2
216  } else if (resbits[0] & EF_RM_TRANSFER_INIT) {
217  ClearEventFlag(event_flag, ~EF_RM_TRANSFER_INIT);
218  SetEventFlag(event_flag, EF_RM_TRANSFER_READY);
219 #ifdef SIO2LOG
220  log_default(LOG_RM_READY);
221 #endif
222  } else if (resbits[0] & EF_UNK_TRANSFER_INIT) {
223  ClearEventFlag(event_flag, ~EF_UNK_TRANSFER_INIT);
224  SetEventFlag(event_flag, EF_UNK_TRANSFER_READY);
225 #ifdef SIO2LOG
226  log_default(LOG_UNK_READY);
227 #endif
228 #endif
229  } else {
230  EPRINTF("Unknown event %08lx. Exiting.\n", resbits[0]);
231  return;
232  }
233 
234 transfer_loop:
235  WaitEventFlag(event_flag, EF_TRANSFER_START | EF_TRANSFER_RESET, 1, resbits);
236 
237  if (resbits[0] & EF_TRANSFER_RESET) {
238  ClearEventFlag(event_flag, ~EF_TRANSFER_RESET);
239 #ifdef SIO2LOG
240  log_default(LOG_RESET);
241 #endif
242  continue;
243  }
244 
245  ClearEventFlag(event_flag, ~EF_TRANSFER_START);
246 
247  sio2_ctrl_set(sio2_ctrl_get() | 0xc);
248  send_td(transfer_data);
249  sio2_ctrl_set(sio2_ctrl_get() | 1);
250 
251  WaitEventFlag(event_flag, EF_SIO2_INTR_COMPLETE, 0, NULL);
252  ClearEventFlag(event_flag, ~EF_SIO2_INTR_COMPLETE);
253 
254  recv_td(transfer_data);
255  SetEventFlag(event_flag, EF_TRANSFER_FINISH);
256 
257  /* Bah... this is needed to get the initial dump from XMCMAN,
258  but it will kill the IOP when XPADMAN is spamming...
259 
260  TODO
261  I guess the correct solution is to do all logging in a
262  dedicated thread. */
263 // log_flush(1);
264 
265  goto transfer_loop;
266  }
267 }

306 int _start(int argc, const char **argv)
307 {
308  int state;
309 
310  shutdown();
311 
312  if (RegisterLibraryEntries(&_exp_sio2man) != 0)
313  return 1;
314 
315  if (init)
316  return 1;
317 
318  init = 1;
319 
320  sio2_ctrl_set(0x3bc);
321 
322  mtap_change_slot_cb = NULL; mtap_get_slot_max_cb = NULL; mtap_get_slot_max2_cb = NULL; mtap_update_slots_cb = NULL;
323  event_flag = create_event_flag();
324  thid = create_main_thread();
325 
326  CpuSuspendIntr(&state);
327  RegisterIntrHandler(IOP_IRQ_SIO2, 1, sio2_intr_handler, &event_flag);
328  EnableIntr(IOP_IRQ_SIO2);
329  CpuResumeIntr(state);
330 
331  dmac_ch_set_dpcr(IOP_DMAC_SIO2in, 3);
332  dmac_ch_set_dpcr(IOP_DMAC_SIO2out, 3);
333  dmac_enable(IOP_DMAC_SIO2in);
334  dmac_enable(IOP_DMAC_SIO2out);
335 
336  StartThread(thid, NULL);
337 #ifdef SIO2LOG
338  EPRINTF("Logging started.\n");
339 #endif
