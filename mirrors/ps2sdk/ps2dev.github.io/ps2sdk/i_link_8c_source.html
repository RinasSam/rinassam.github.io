251 int iLinkSetNodeTrSpeed(int trContext, unsigned char speed){
252  int result;
253 
254  result=iLinkGetNodeTrSpeed(trContext);
255 
256  /* !!HACK!! Due to a chip bug (?) in the iLink hardware of most consoles, limit the speed to S100 mode whenever the console is the root node
257  (Read the README file for more details on this bug). */
258  if(IsBusRoot) speed=S100;
259 
260  /* Assign the specified speed to the specified transaction. Assign the maxium allowed speed if the specified speed is above the maximum value. */
261  TransactionContexts[trContext].speed=(speed>(sizeof(speeds)-1))?speeds[sizeof(speeds)-1]:speeds[speed];
262 
263  return result;
264 }
