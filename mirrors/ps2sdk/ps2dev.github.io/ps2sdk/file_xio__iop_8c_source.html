128 static int fileXio_GetDeviceList_RPC(struct fileXioDevice* ee_devices, int eecount)
129 {
130  int device_count = 0;
131  iop_device_t **devices = GetDeviceList();
132  struct fileXioDevice local_devices[FILEXIO_MAX_DEVICES];
133  while (devices[device_count] && device_count < eecount)
134  {
135  iop_device_t *device = devices[device_count];
136  strncpy(local_devices[device_count].name, device->name, 128);
137  local_devices[device_count].name[15] = '\0';
138  local_devices[device_count].type = device->type;
139  local_devices[device_count].version = device->version;
140  strncpy(local_devices[device_count].desc, device->desc, 128);
141  local_devices[device_count].desc[127] = '\0';
142  ++device_count;
143  }
144  if (device_count)
145  {
146  struct t_SifDmaTransfer dmaStruct;
147  int intStatus; // interrupt status - for dis/en-abling interrupts
148 
149  dmaStruct.src = local_devices;
150  dmaStruct.dest = ee_devices;
151  dmaStruct.size = sizeof(struct fileXioDevice) * device_count;
152  dmaStruct.attr = 0;
153 
154  // Do the DMA transfer
155  CpuSuspendIntr(&intStatus);
156  SifSetDma(&dmaStruct, 1);
157  CpuResumeIntr(intStatus);
158  }
159  return device_count;
160 }

162 static int fileXio_CopyFile_RPC(const char *src, const char *dest, int mode)
163 {
164  iox_stat_t stat;
165  int infd, outfd, size, remain, i, retval = 0;
166 
167  if ((infd = open(src, O_RDONLY, 0666)) < 0) {
168  return infd;
169  }
170  if ((outfd = open(dest, O_RDWR|O_CREAT|O_TRUNC, 0666)) < 0) {
171  close(infd);
172  return outfd;
173  }
174  size = lseek(infd, 0, SEEK_END);
175  lseek(infd, 0, SEEK_SET);
176  if (!size)
177  return retval;
178 
179  remain = size % RWBufferSize;
180  for (i = 0; i < (size / RWBufferSize); i++) {
181  read(infd, rwbuf, RWBufferSize);
182  write(outfd, rwbuf, RWBufferSize);
183  }
184  read(infd, rwbuf, remain);
185  write(outfd, rwbuf, remain);
186  close(infd);
187  close(outfd);
188 
189  stat.mode = mode;
190  chstat(dest, &stat, 1);
191 
192  return size;
193 }

195 static int fileXio_Read_RPC(int infd, char *read_buf, int read_size, void *intr_data)
196 {
197  int srest;
198  int erest;
199  int asize;
200  int abuffer;
201  int rlen;
202  int total;
203  int readlen;
204  int status;
205  struct t_SifDmaTransfer dmaStruct;
206  void *buffer;
207  void *aebuffer;
208  int intStatus; // interrupt status - for dis/en-abling interrupts
209  int read_buf2 = (int)read_buf;
210 
211  total = 0;
212 
213  if(read_size < 64)
214  {
215  srest = read_size;
216  erest = asize = abuffer = 0;
217  buffer = read_buf;
218  aebuffer = 0;
219  }else{
220  if ((read_buf2 & 0x3F) == 0)
221  srest=0;
222  else
223  srest=(int)RDOWN_64(read_buf2) - read_buf2 + 64;
224  buffer = read_buf;
225  abuffer = read_buf2 + srest;
226  aebuffer=(void *)RDOWN_64(read_buf2 + read_size);
227  asize = (int)aebuffer - (int)abuffer;
228  erest = (read_buf2 + read_size) - (int)aebuffer;
229  }
230  if (srest>0)
231  {
232  if (srest!=(rlen=read(infd, rests.sbuffer, srest)))
233  {
234  total += srest = (rlen>0 ? rlen:0);
235  goto EXIT;
236  }
237  else
238  total += srest;
239  }
240 
241  status=0;
242  while (asize>0)
243  {
244  readlen=MIN(RWBufferSize, asize);
245 
246  while(SifDmaStat(status)>=0);
247 
248  rlen=read(infd, rwbuf, readlen);
249  if (readlen!=rlen){
250  if (rlen<=0)goto EXIT;
251  dmaStruct.dest=(void *)abuffer;
252  dmaStruct.size=rlen;
253  dmaStruct.attr=0;
254  dmaStruct.src =rwbuf;
255  CpuSuspendIntr(&intStatus);
256  SifSetDma(&dmaStruct, 1);
257  CpuResumeIntr(intStatus);
258  total +=rlen;
259  goto EXIT;
260  }else{ //read ok
261  total += rlen;
262  asize -=rlen;
263  dmaStruct.dest=(void *)abuffer;
264  abuffer +=rlen;
265  dmaStruct.size=rlen;
266  dmaStruct.attr=0;
267  dmaStruct.src =rwbuf;
268  CpuSuspendIntr(&intStatus);
269  status=SifSetDma(&dmaStruct, 1);
270  CpuResumeIntr(intStatus);
271  }
272  }
273  if (erest>0)
274  {
275  rlen = read(infd, rests.ebuffer, erest);
276  total += (rlen>0 ? rlen : 0);
277  }
278 EXIT:
279  rests.ssize=srest;
280  rests.esize=erest;
281  rests.sbuf =buffer;
282  rests.ebuf =aebuffer;
283  dmaStruct.src =&rests;
284  dmaStruct.size=sizeof(rests_pkt);
285  dmaStruct.attr=0;
286  dmaStruct.dest=intr_data;
287  CpuSuspendIntr(&intStatus);
288  SifSetDma(&dmaStruct, 1);
289  CpuResumeIntr(intStatus);
290  return (total);
291 }

294 {
295  SifRpcReceiveData_t rdata;
296  int left;
297  int wlen;
298  int writelen;
299  int pos;
300  int total;
301 
302  left = write_size;
303  total = 0;
304  if (mis > 0)
305  {
306  wlen=write(outfd, misbuf, mis);
307  if (wlen != mis)
308  {

312  }
313  total += wlen;
314  }
315 
316  left-=mis;
317  pos=(int)write_buf+mis;
318  while(left){

323  if (wlen>0)
324  total+=wlen;
325  return (total);
326  }
327  left -=writelen;
328  pos +=writelen;
329  total+=writelen;
330  }
331  return (total);
332 }

337 static int fileXio_GetDir_RPC(const char* pathname, struct fileXioDirEntry dirEntry[], unsigned int req_entries)
338 {
339  int matched_entries;
340  int fd, res;
341  iox_dirent_t dirbuf;
342  struct fileXioDirEntry localDirEntry;
343  int intStatus; // interrupt status - for dis/en-abling interrupts
344  struct t_SifDmaTransfer dmaStruct;
345  int dmaID;
346 
347  dmaID = 0;
348 
349  #ifdef DEBUG
350  printf("RPC GetDir Request\n");
351  printf("dirname: %s\n",pathname);
352  #endif
353 
354  matched_entries = 0;
355 
356  fd = dopen(pathname);
357  if (fd <= 0)
358  {
359  return fd;
360  }
361  {
362 
363  res = 1;
364  while (res > 0)
365  {
366  memset(&dirbuf, 0, sizeof(dirbuf));
367  res = dread(fd, &dirbuf);
368  if (res > 0)
369  {
370  // check for too many entries
371  if (matched_entries == req_entries)
372  {
373  close(fd);
374  return (matched_entries);
375  }
376  // wait for any previous DMA to complete
377  // before over-writing localDirEntry
378  while(SifDmaStat(dmaID)>=0);
379  DirEntryCopy(&localDirEntry, &dirbuf);
380  // DMA localDirEntry to the address specified by dirEntry[matched_entries]
381  // setup the dma struct
382  dmaStruct.src = &localDirEntry;
383  dmaStruct.dest = &dirEntry[matched_entries];
384  dmaStruct.size = sizeof(struct fileXioDirEntry);
385  dmaStruct.attr = 0;
386  // Do the DMA transfer
387  CpuSuspendIntr(&intStatus);
388  dmaID = SifSetDma(&dmaStruct, 1);
389  CpuResumeIntr(intStatus);
390  matched_entries++;
391  } // if res
392  } // while
393 
394  } // if dirs and files
395 
396  // cleanup and return # of entries
397  close(fd);
398  return (matched_entries);
399 }
