183  if (arg->isActive)
184  return -1;
185 
186  arg->isActive = 1;
187  arg->callbackProc = func;
188  arg->callbackArg = cbArg;
189 
190  TimerCbStruct *pos = memPool.timerListStart;
191 
192  while (pos && (delay > pos->delayCount)) {
193  delay -= pos->delayCount;

195  }
196  if (pos) {
197  arg->prev = pos->prev;

199  pos->prev->next = arg;
200  else

206  } else {

208  if (memPool.timerListEnd)
209  memPool.timerListEnd->next = arg;
210  else

212  memPool.timerListEnd = arg;

386 void fetchConfigDescriptors(IoRequest *req) {
387  Endpoint *ep = req->correspEndpoint;
388  Device *dev = ep->correspDevice;
389  u16 readLen;
390  int fetchDesc;
391 
392  if ((req->resultCode == USB_RC_OK) || (dev->fetchDescriptorCounter == 0)) {
393  u32 curDescNum = dev->fetchDescriptorCounter++;
394 
395  fetchDesc = curDescNum & 1;
396  curDescNum >>= 1;
397 
398  if ((curDescNum > 0) && !fetchDesc) {
399  UsbConfigDescriptor *desc = dev->staticDeviceDescEndPtr;
400  dev->staticDeviceDescEndPtr += READ_UINT16(&desc->wTotalLength);
401  }
402 
403  if (fetchDesc) {
404  UsbConfigDescriptor *desc = dev->staticDeviceDescEndPtr;
405  readLen = READ_UINT16(&desc->wTotalLength);
406  } else
407  readLen = 4;
408 
409  if ((u8*)dev->staticDeviceDescEndPtr + readLen > (u8*)dev->staticDeviceDescPtr + usbConfig.maxStaticDescSize) {
410  dbg_printf("USBD: Device ignored, Device descriptors too large\n");
411  return; // buffer is too small, silently ignore the device
412  }
413 
414  if (curDescNum < ((UsbDeviceDescriptor*)dev->staticDeviceDescPtr)->bNumConfigurations) {
415  doControlTransfer(ep, &dev->ioRequest,
416  USB_DIR_IN | USB_RECIP_DEVICE, USB_REQ_GET_DESCRIPTOR, (USB_DT_CONFIG << 8) | curDescNum, 0, readLen,
417  dev->staticDeviceDescEndPtr, fetchConfigDescriptors);
418  } else
419  connectNewDevice(dev);
420  } else
421  killDevice(dev, ep);

506  feature = C_PORT_CONNECTION;
507  else if (dev->portStatusChange & BIT(C_PORT_ENABLE))
508  feature = C_PORT_ENABLE;
509  else if (dev->portStatusChange & BIT(C_PORT_SUSPEND))
510  feature = C_PORT_SUSPEND;
511  else if (dev->portStatusChange & BIT(C_PORT_OVER_CURRENT))
512  feature = C_PORT_OVER_CURRENT;
513  else if (dev->portStatusChange & BIT(C_PORT_RESET))
514  feature = C_PORT_RESET;
515 
516  if (feature >= 0) {
517  dev->portStatusChange &= ~BIT(feature);
518  HubControlTransfer(dev,

520  hubGetPortStatusCallback);

523  if (port) {

529  addTimerCallback(&port->timer, (TimerCallback)hubResetDevice, port, 500);
530  return;

547  }
