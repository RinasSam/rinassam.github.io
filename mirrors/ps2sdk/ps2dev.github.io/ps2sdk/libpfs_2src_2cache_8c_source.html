30 void pfsCacheFree(pfs_cache_t *clink)
31 {
32  if(clink==NULL) {
33  PFS_PRINTF(PFS_DRV_NAME": Warning: NULL buffer returned\n");
34  return;
35  }
36 
37  if(clink->nused==0){
38  PFS_PRINTF(PFS_DRV_NAME": Error: Unused cache returned\n");
39  return;
40  }
41 
42  clink->nused--;
43  if(clink->pfsMount!=NULL) {
44  if(clink->nused!=0)
45  return;
46  pfsCacheLink(pfsCacheBuf->prev, clink);
47  return;
48  }
49  if(clink->nused!=0) {
50  PFS_PRINTF(PFS_DRV_NAME": Warning: Invalidated buffer is in use\n");
51  return;
52  }
53  pfsCacheLink(pfsCacheBuf, clink);
54 }

77 int pfsCacheTransfer(pfs_cache_t* clink, int mode)
78 {
79  pfs_mount_t *pfsMount=clink->pfsMount;
80  int err;
81 
82  if(pfsMount->lastError == 0) { // no error
83  if((err=pfsMount->blockDev->transfer(pfsMount->fd, clink->u.data, clink->sub,
84  clink->block << pfsBlockSize, 1 << pfsBlockSize, mode))==0) {
85  if(mode==PFS_IO_MODE_READ) {
86  if(clink->flags & PFS_CACHE_FLAG_SEGD && ((pfs_inode_t *)clink->u.inode)->magic!=PFS_SEGD_MAGIC)
87  err=-EIO;
88  if(clink->flags & PFS_CACHE_FLAG_SEGI && ((pfs_inode_t *)clink->u.inode)->magic!=PFS_SEGI_MAGIC)
89  err=-EIO;
90  if(clink->flags & (PFS_CACHE_FLAG_SEGD|PFS_CACHE_FLAG_SEGI)) {
91  if(((pfs_inode_t *)clink->u.inode)->checksum!=pfsInodeCheckSum(clink->u.inode))
92  err=-EIO;
93  }
94  }
95  }
96  if(err!=0) {
97  PFS_PRINTF(PFS_DRV_NAME": Error: Disk error partition %ld, block %ld, err %d\n",
98  clink->sub, clink->block, err);
99 #ifndef PFS_NO_WRITE_ERROR_STAT
100  pfsMount->blockDev->setPartitionError(pfsMount->fd);
101  pfsFsckStat(pfsMount, clink->u.superblock, PFS_FSCK_STAT_WRITE_ERROR, PFS_MODE_SET_FLAG);
102  pfsMount->lastError=err;
103 #endif
104  }
105  }
106  clink->flags&=~PFS_CACHE_FLAG_DIRTY; // clear dirty :)
107  return pfsMount->lastError;
108 }

110 void pfsCacheFlushAllDirty(pfs_mount_t *pfsMount)
111 {
112  u32 i;
113  int found=0;
114 
115  for(i=1;i<pfsCacheNumBuffers+1;i++){
116  if(pfsCacheBuf[i].pfsMount == pfsMount &&
117  pfsCacheBuf[i].flags & PFS_CACHE_FLAG_DIRTY)
118  found=1;
119  }
120  if(found) {
121  pfsJournalWrite(pfsMount, pfsCacheBuf+1, pfsCacheNumBuffers);
122  for(i=1;i<pfsCacheNumBuffers+1;i++){
123  if(pfsCacheBuf[i].pfsMount == pfsMount &&
124  pfsCacheBuf[i].flags & PFS_CACHE_FLAG_DIRTY)
125  pfsCacheTransfer(&pfsCacheBuf[i], 1);
126  }
127  }
128 
129  pfsJournalReset(pfsMount);

132 pfs_cache_t *pfsCacheAlloc(pfs_mount_t *pfsMount, u16 sub, u32 block,
133  int flags, int *result)
134 {
135  pfs_cache_t *allocated;
136 
137  if (pfsCacheBuf->prev==pfsCacheBuf && pfsCacheBuf->prev->next==pfsCacheBuf->prev) {
138  PFS_PRINTF(PFS_DRV_NAME": Error: Free buffer list is empty\n");
139  *result=-ENOMEM;
140  return NULL;
141  }
142  allocated=pfsCacheBuf->next;
143  if (pfsCacheBuf->next==NULL)
144  PFS_PRINTF(PFS_DRV_NAME": Panic: Null pointer allocated\n");
145  if (allocated->pfsMount && (allocated->flags & PFS_CACHE_FLAG_DIRTY))
146  pfsCacheFlushAllDirty(allocated->pfsMount);
147  allocated->flags = flags & PFS_CACHE_FLAG_MASKTYPE;
148  allocated->pfsMount = pfsMount;
149  allocated->sub = sub;
150  allocated->block = block;
151  allocated->nused = 1;
152  return pfsCacheUnLink(allocated);
153 }

155 pfs_cache_t *pfsCacheGetData(pfs_mount_t *pfsMount, u16 sub, u32 block,
156  int flags, int *result)
157 {
158  u32 i;
159  pfs_cache_t *clink;
160 
161  *result=0;
162 
163  for (i=1; i < pfsCacheNumBuffers + 1; i++)
164  if ( pfsCacheBuf[i].pfsMount &&
165  (pfsCacheBuf[i].pfsMount==pfsMount) &&
166  (pfsCacheBuf[i].block == block))
167  if (pfsCacheBuf[i].sub==sub){
168  pfsCacheBuf[i].flags &= PFS_CACHE_FLAG_MASKSTATUS;
169  pfsCacheBuf[i].flags |= flags & PFS_CACHE_FLAG_MASKTYPE;
170  if (pfsCacheBuf[i].nused == 0)
171  pfsCacheUnLink(&pfsCacheBuf[i]);
172  pfsCacheBuf[i].nused++;
173  return &pfsCacheBuf[i];
174  }
175 
176  clink=pfsCacheAlloc(pfsMount, sub, block, flags, result);
177 
178  if (clink){
179  if (flags & PFS_CACHE_FLAG_NOLOAD)
180  return clink;
181 
182  if ((*result=pfsCacheTransfer(clink, PFS_IO_MODE_READ))>=0)
183  return clink;
184 
185  clink->pfsMount=NULL;
186  pfsCacheFree(clink);
187  }
188  return NULL;
189 }

204 int pfsCacheInit(u32 numBuf, u32 bufSize)
205 {
206  char *cacheData;
207  u32 i;
208 
209  if(numBuf > 127) {
210  PFS_PRINTF(PFS_DRV_NAME": Error: Number of buffers larger than 127.\n");
211  return -EINVAL;
212  }
213 
214  cacheData = pfsAllocMem(numBuf * bufSize);
215 
216  if(!cacheData || !(pfsCacheBuf = pfsAllocMem((numBuf + 1) * sizeof(pfs_cache_t))))
217  return -ENOMEM;
218 
219  pfsCacheNumBuffers = numBuf;
220  memset(pfsCacheBuf, 0, (numBuf + 1) * sizeof(pfs_cache_t));
221 
222  pfsCacheBuf->next = pfsCacheBuf;
223  pfsCacheBuf->prev = pfsCacheBuf;
224 
225  for(i = 1; i < numBuf + 1; i++)
226  {
227  pfsCacheBuf[i].u.data = cacheData;
228  pfsCacheLink(pfsCacheBuf->prev, &pfsCacheBuf[i]);
229  cacheData += bufSize;
230  }
231 
232  return 0;
233 }
