64 Endpoint *openDeviceEndpoint(Device *dev, UsbEndpointDescriptor *endpDesc, u32 alignFlag) {
65 
66  u16 flags = 0;
67  u16 hcMaxPktSize;
68  u8 endpType = 0;
69  u8 type;
70  HcTD *td = NULL;
71 
72  Endpoint *newEp = allocEndpointForDevice(dev, alignFlag);
73 
74  if (!newEp) {
75  dbg_printf("ran out of endpoints\n");
76  return NULL;
77  }
78 
79  if (endpDesc) {
80  type = endpDesc->bmAttributes & USB_ENDPOINT_XFERTYPE_MASK;
81  hcMaxPktSize = (endpDesc->wMaxPacketSizeHB << 8) | endpDesc->wMaxPacketSizeLB;
82 
83  if (type == USB_ENDPOINT_XFER_ISOC) {
84  endpType = TYPE_ISOCHRON;
85  td = (HcTD *)allocIsoTd();
86  if (!td) {
87  cleanUpFunc(dev, newEp);
88  dbg_printf("Open ISOC EP: no TDs left\n");
89  return NULL;
90  }
91  } else if (type == USB_ENDPOINT_XFER_CONTROL) {
92  endpType = TYPE_CONTROL;
93  if (!alignFlag)
94  if (hcMaxPktSize >= 0x3F)
95  hcMaxPktSize = 0x3E;
96  } else { // BULK or INT
97  if (type == USB_ENDPOINT_XFER_INT) {
98  if (endpDesc->bInterval >= 0x20)
99  endpType = 0x1F;
100  else if (endpDesc->bInterval >= 0x10)
101  endpType = 0xF;
102  else if (endpDesc->bInterval >= 8)
103  endpType = 7;
104  else if (endpDesc->bInterval >= 4)
105  endpType = 3;
106  else if (endpDesc->bInterval >= 2)
107  endpType = 1;
108  else
109  endpType = 0;
110 
111  // todo: add bandwidth scheduling
112 
113  dbg_printf("opening INT endpoint (%d - %p), interval %d, list %d\n", newEp->id, newEp, endpDesc->bInterval, endpType);
114  } else
115  endpType = TYPE_BULK;
116 
117  if (((endpDesc->bEndpointAddress & USB_DIR_IN) == 0) && !alignFlag)
118  if (hcMaxPktSize >= 0x3F)
119  hcMaxPktSize = 0x3E;
120  }
121 
122  hcMaxPktSize &= 0x7FF;
123  if (type == USB_ENDPOINT_XFER_ISOC)
124  flags |= HCED_ISOC;
125  if (dev->isLowSpeedDevice)
126  flags |= HCED_SPEED;
127 
128  flags |= (endpDesc->bEndpointAddress & 0xF) << 7;
129 
130  if (endpDesc->bEndpointAddress & USB_DIR_IN)
131  flags |= HCED_DIR_IN;
132  else
133  flags |= HCED_DIR_OUT;
134 
135  flags |= dev->functionAddress & 0x7F;
136 
137  newEp->hcEd.hcArea = flags;
138  newEp->hcEd.maxPacketSize = hcMaxPktSize;
139  } else {
140  newEp->hcEd.maxPacketSize = 8;
141  if (dev->isLowSpeedDevice)
142  newEp->hcEd.hcArea = HCED_SPEED;
143  else
144  newEp->hcEd.hcArea = 0;
145  endpType = TYPE_CONTROL;
146  }
147 
148  newEp->endpointType = endpType;
149  if (!td) {
150  td = allocTd();
151  if (!td) {
152  dbg_printf("Ran out of TDs\n");
153  cleanUpFunc(dev, newEp);
154  return NULL;
155  }
156  }
157  newEp->hcEd.tdHead = newEp->hcEd.tdTail = td;
158  addToHcEndpointList(newEp->endpointType, &newEp->hcEd);
159  return newEp;
160 }

203 void handleRhsc(void) {
204  u32 portNum = 0;
205  Device *port = memPool.deviceTreeRoot->childListStart;
206 
207  while (port) {
208  u32 status = memPool.ohciRegs->HcRhPortStatus[portNum];
209  memPool.ohciRegs->HcRhPortStatus[portNum] = C_PORT_FLAGS; // reset all flags
210  if (status & BIT(PORT_CONNECTION)) {
211  if ((port->deviceStatus != DEVICE_NOTCONNECTED) && (status & BIT(C_PORT_CONNECTION)))
212  flushPort(port);
213 
214  if (port->deviceStatus == DEVICE_NOTCONNECTED) {
215  port->deviceStatus = DEVICE_CONNECTED;
216  addTimerCallback(&port->timer, (TimerCallback)hubResetDevice, port, 500);
217  } else if (port->deviceStatus == DEVICE_RESETPENDING) {
218  if (!(status & BIT(PORT_RESET))) {
219  port->deviceStatus = DEVICE_RESETCOMPLETE;
220  port->isLowSpeedDevice = (status >> PORT_LOW_SPEED) & 1;
221  Endpoint *ep = openDeviceEndpoint(port, NULL, 0);
222  if (ep)
223  hubTimedSetFuncAddress(port);
224  }
225  }
226  } else
227  flushPort(port);
228  port = port->next;
229  portNum++;
230  }
231 }

233 void hcdProcessIntr(void) {
234  u32 intrFlags;
235 
236  intrFlags = memPool.ohciRegs->HcInterruptStatus & memPool.ohciRegs->HcInterruptEnable;
237 
238  if (intrFlags & OHCI_INT_SO) {
239  dbg_printf("HC: Scheduling overrun\n");
240  memPool.ohciRegs->HcInterruptStatus = OHCI_INT_SO;
241  intrFlags &= ~OHCI_INT_SO;
242  }
243 
244  HcTD *doneQueue = (HcTD*)((u32)memPool.hcHCCA->DoneHead & ~0xF);
245  if (doneQueue) {
246  memPool.hcHCCA->DoneHead = NULL;
247  memPool.ohciRegs->HcInterruptStatus = OHCI_INT_WDH;
248 
249  // reverse queue
250  HcTD *prev = NULL;
251  do {
252  HcTD *tmp = doneQueue;
253  doneQueue = tmp->next;
254  tmp->next = prev;
255  prev = tmp;
256  } while (doneQueue);
257 
258  do {
259  HcTD *tmp = prev->next;
260  if ((prev >= memPool.hcTdBuf) && (prev < memPool.hcTdBufEnd))
261  processDoneQueue_GenTd(prev);
262  else if ((prev >= (HcTD *)memPool.hcIsoTdBuf) && (prev < (HcTD *)memPool.hcIsoTdBufEnd))
263  processDoneQueue_IsoTd((HcIsoTD *)prev);
264  prev = tmp;
265  } while (prev);
266 
267  intrFlags &= ~OHCI_INT_WDH;
268  }
269 
270  if (intrFlags & OHCI_INT_SF) {
271  memPool.ohciRegs->HcInterruptStatus = OHCI_INT_SF;
272  handleTimerList();
273  intrFlags &= ~OHCI_INT_SF;
274  }
275 
276  if (intrFlags & OHCI_INT_UE) {
277  printf("HC: Unrecoverable error\n");
278  memPool.ohciRegs->HcInterruptStatus = OHCI_INT_UE;
279  intrFlags &= ~OHCI_INT_UE;
280  }
281 
282  if (intrFlags & OHCI_INT_RHSC) {
283  dbg_printf("RHSC\n");
284  memPool.ohciRegs->HcInterruptStatus = OHCI_INT_RHSC;
285  handleRhsc();
286  intrFlags &= ~OHCI_INT_RHSC;
287  }
288 
289  intrFlags &= ~OHCI_INT_MIE;
290  if (intrFlags) {
291  dbg_printf("Disable intr: %d\n", intrFlags);
292  memPool.ohciRegs->HcInterruptDisable = intrFlags;
293  }
