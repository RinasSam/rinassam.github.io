329 void adddate(sceCdCLOCK* time)
330 {
331  // get the days in each month and fix up feb depending on leap year
332  unsigned char days_in_months[12];
333  memcpy(days_in_months, gDaysInMonths, 12);
334  if((time->year & 3) == 0)
335  days_in_months[1] = 29;
336 
337  // increment the day and check its within the "day of the month" bounds
338  time->day++;
339  if(time->day > days_in_months[time->month - 1])
340  {
341  time->day = 1;
342 
343  // increment the month and check its within the "months in a year" bounds
344  time->month++;
345  if(time->month == 13)
346  {
347  time->month = 1;
348 
349  // check the year and increment it
350  time->year++;
351  if(time->year == 100)
352  {
353  time->year = 0;
354  }
355  }
356  }
357 }
358 void subdate(sceCdCLOCK* time)
359 {
360  // get the days in each month and fix up feb depending on leap year
361  unsigned char days_in_months[12];
362  memcpy(days_in_months, gDaysInMonths, 12);
363  if((time->year & 3) == 0)
364  days_in_months[1] = 29;
365 
366  // decrement the day and check its within the "day of the month" bounds
367  time->day--;
368  if(time->day == 0)
369  {
370  // decrement the month and check its within the "months in a year" bounds
371  time->month--;
372  if(time->month == 0)
373  {
374  time->month = 12;
375 
376  // check the year and decrement it
377  if(time->year == 0)
378  time->year = 99;
379  else
380  time->year--;
381  }
382 
383  time->day = days_in_months[time->month-1];
