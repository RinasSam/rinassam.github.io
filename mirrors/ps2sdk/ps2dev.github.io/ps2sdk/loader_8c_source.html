67 int main(int argc, char *argv[])
68 {
69  static t_ExecData elfdata;
70  int ret, i;
71 
72  elfdata.epc = 0;
73 
74  GS_BGCOLOUR = WHITE_BG;
75  // arg[0] partition if exists, otherwise is ""
76  // arg[1]=path to ELF
77  if (argc < 2) {
78  GS_BGCOLOUR = RED_BG;
79  return -EINVAL;
80  }
81 
82  char *new_argv[argc - 1];
83  int fullPath_length = 1 + strlen(argv[0]) + strlen(argv[1]);
84  char fullPath[fullPath_length];
85  strcpy(fullPath, argv[0]);
86  strcat(fullPath, argv[1]);
87  // final new_argv[0] is partition + path to elf
88  new_argv[0] = fullPath;
89  for (i = 2; i < argc; i++) {
90  new_argv[i - 1] = argv[i];
91  }
92 
93  GS_BGCOLOUR = CYAN_BG;
94 
95  // Initialize
96  SifInitRpc(0);
97  wipeUserMem();
98 
99  //Writeback data cache before loading ELF.
100  FlushCache(0);
101  GS_BGCOLOUR = GREEN_BG;
102  SifLoadFileInit();
103  ret = SifLoadElf(argv[1], &elfdata);
104  SifLoadFileExit();
105  GS_BGCOLOUR = BLUE_BG;
106  if (ret == 0 && elfdata.epc != 0) {
107  GS_BGCOLOUR = YELLOW_BG;
108 
109  // Let's reset IOP because ELF was already loaded in memory
110  while(!SifIopReset(NULL, 0)){};
111  while (!SifIopSync()) {};
112 
113  GS_BGCOLOUR = RED_BG;
114 
115  SifInitRpc(0);
116  // Load modules.
117  SifLoadFileInit();
118  SifLoadModule("rom0:SIO2MAN", 0, NULL);
119  SifLoadModule("rom0:MCMAN", 0, NULL);
120  SifLoadModule("rom0:MCSERV", 0, NULL);
121  SifLoadFileExit();
122  SifExitRpc();
123 
124  GS_BGCOLOUR = BROWN_BG;
125 
126  FlushCache(0);
127  FlushCache(2);
128 
129  GS_BGCOLOUR = PURPBLE_BG;
130 
131  return ExecPS2((void *)elfdata.epc, (void *)elfdata.gp, argc-1, new_argv);
132  } else {
133  GS_BGCOLOUR = MAGENTA_BG;
134  SifExitRpc();
135  return -ENOENT;
136  }
137 }
