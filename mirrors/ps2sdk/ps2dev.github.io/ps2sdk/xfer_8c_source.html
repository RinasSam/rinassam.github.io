82

84 
85  /* Non-Sony: Workaround for the hardware BUG whereby the Rx FIFO of the MAL becomes unresponsive or loses frames when under load.
86  Check that there are frames to process, before accessing the BD registers. */
87  while (SMAP_REG8(SMAP_R_RXFIFO_FRAME_CNT) > 0) {
88  PktBdPtr = &rx_bd[SmapDrivPrivData->RxBDIndex % SMAP_BD_MAX_ENTRY];
89  ctrl_stat = PktBdPtr->ctrl_stat;
90  if (!(ctrl_stat & SMAP_BD_RX_EMPTY)) {
91  length = PktBdPtr->length;
92  LengthRounded = (length + 3) & ~3;
93  pointer = PktBdPtr->pointer;
94 
95  if (ctrl_stat & (SMAP_BD_RX_INRANGE | SMAP_BD_RX_OUTRANGE | SMAP_BD_RX_FRMTOOLONG | SMAP_BD_RX_BADFCS | SMAP_BD_RX_ALIGNERR | SMAP_BD_RX_SHORTEVNT | SMAP_BD_RX_RUNTFRM | SMAP_BD_RX_OVERRUN)) {
96  for (i = 0; i < 16; i++)
97  if ((ctrl_stat >> i) & 1)
98  SmapDrivPrivData->RuntimeStats.RxErrorCount++;
99 
100  SmapDrivPrivData->RuntimeStats.RxDroppedFrameCount++;
101 
102  if (ctrl_stat & SMAP_BD_RX_OVERRUN)
103  SmapDrivPrivData->RuntimeStats.RxFrameOverrunCount++;
104  if (ctrl_stat & (SMAP_BD_RX_INRANGE | SMAP_BD_RX_OUTRANGE | SMAP_BD_RX_FRMTOOLONG | SMAP_BD_RX_SHORTEVNT | SMAP_BD_RX_RUNTFRM))
105  SmapDrivPrivData->RuntimeStats.RxFrameBadLengthCount++;
106  if (ctrl_stat & SMAP_BD_RX_BADFCS)
107  SmapDrivPrivData->RuntimeStats.RxFrameBadFCSCount++;
108  if (ctrl_stat & SMAP_BD_RX_ALIGNERR)
109  SmapDrivPrivData->RuntimeStats.RxFrameBadAlignmentCount++;
110 
111  // Original did this whenever a frame is dropped.
112  SMAP_REG16(SMAP_R_RXFIFO_RD_PTR) = pointer + LengthRounded;
113  } else {
114  if ((pbuf = NetManNetProtStackAllocRxPacket(LengthRounded, &payload)) != NULL) {
115  CopyFromFIFO(SmapDrivPrivData->smap_regbase, payload, length, pointer);
116  NetManNetProtStackEnQRxPacket(pbuf);
117  NumPacketsReceived++;
118  } else {
119  SmapDrivPrivData->RuntimeStats.RxAllocFail++;
120  // Original did this whenever a frame is dropped.
121  SMAP_REG16(SMAP_R_RXFIFO_RD_PTR) = pointer + LengthRounded;
122  }
123  }
124 
125  SMAP_REG8(SMAP_R_RXFIFO_FRAME_DEC) = 0;
126  PktBdPtr->ctrl_stat = SMAP_BD_RX_EMPTY;
127  SmapDrivPrivData->RxBDIndex++;

163

169  SmapDrivPrivData->TxBDIndex++;

175  printf("smap: dropped\n");
