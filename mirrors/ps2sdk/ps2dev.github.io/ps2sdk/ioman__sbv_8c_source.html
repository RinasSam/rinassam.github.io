39 {

71 void fix_imports(iop_library_t *lib)
72 {
73  struct irx_import_table *table;
74  struct irx_import_stub *stub;
75 
76  FlushDcache();
77 
78  // go through each table that imports the library
79  for(table = lib->caller; table != NULL; table = table->next)
80  {
81  // go through each import in the table
82  for(stub = (struct irx_import_stub *) table->stubs; stub->jump != 0; stub++)
83  {
84  // patch the stub to jump to the address specified in the library export table for "fno"
85  stub->jump = 0x08000000 | (((u32) lib->exports[stub->fno] << 4) >> 6);
86  }
87  }
88 
89  FlushIcache();
90 }

191 int ioman_dread(int fd, io_dirent_t *io_dirent)
192 {
193  iop_file_t *f = get_file(fd);
194  int res;
195 
196  if (f == NULL || !(f->mode & 8))
197  return -EBADF;
198 
199  /* If this is a new device (such as pfs:) then we need to convert the mode
200  variable of the stat structure to ioman's format. */
201  if ((f->device->type & 0xf0000000) == IOP_DT_FSEXT)
202  {
203  iox_dirent_t iox_dirent;
204  res = f->device->ops->dread(f, &iox_dirent);
205 
206  statx2stat(&iox_dirent.stat, &io_dirent->stat);
207 
208  strncpy(io_dirent->name, iox_dirent.name, sizeof(iox_dirent.name));
209  }
210  else
211  {
212  typedef int io_dread_t(iop_file_t *, io_dirent_t *);
213  io_dread_t *io_dread = (io_dread_t*) f->device->ops->dread;
214  res = io_dread(f, io_dirent);
215  }
216 
217  return res;
218 }
